## NodeHandlers

### validate-manifest

```typescript
async function validateManifest(
  ctx: WorkflowContext,
  input: CleanupManifestInput
): Promise<{ manifest: Manifest; canCleanup: boolean }> {
  const manifest = queryOne<Manifest>(
    'SELECT * FROM manifests WHERE id = ?',
    [input.manifestId]
  );

  if (!manifest) {
    throw new NotFoundError(`Manifest ${input.manifestId} not found`);
  }

  if (manifest.status !== 'SEALED') {
    throw new ValidationError('Cannot cleanup DRAFT manifest');
  }

  const now = Date.now();
  if (!input.force && manifest.expires_at && manifest.expires_at > now) {
    throw new ValidationError('Manifest not expired yet (use force=true to override)');
  }

  return { manifest, canCleanup: true };
}
```

### load-resources

```typescript
async function loadResources(
  ctx: WorkflowContext,
  _input: CleanupManifestInput,
  deps: { 'validate-manifest': { manifest: Manifest } }
): Promise<{ resourcesByPriority: Map<number, ManifestResource[]> }> {
  const { manifest } = deps['validate-manifest'];

  const resources = queryMany<ManifestResource>(
    'SELECT * FROM manifest_resources WHERE manifest_id = ? ORDER BY cleanup_priority DESC',
    [manifest.id]
  );

  // Group by priority
  const resourcesByPriority = new Map<number, ManifestResource[]>();
  for (const resource of resources) {
    const list = resourcesByPriority.get(resource.cleanup_priority) || [];
    list.push(resource);
    resourcesByPriority.set(resource.cleanup_priority, list);
  }

  // Per-provider scoping: group resources by provider (spec Ch04, Ch07 MUST requirement).
  // Cleanup of each provider's resources proceeds independently so cross-provider
  // failures don't block cleanup of unrelated providers.
  const resourcesByProvider = new Map<string, ManifestResource[]>();
  for (const resource of resources) {
    // Resolve provider from the resource's backing instance or object
    const provider = await resolveResourceProvider(resource);
    const list = resourcesByProvider.get(provider) || [];
    list.push(resource);
    resourcesByProvider.set(provider, list);
  }

  return { resourcesByPriority, resourcesByProvider };
}
```

### resolveResourceProvider (helper)

```typescript
/**
 * Resolve the provider for a manifest resource by looking up the backing
 * instance or object. Used for per-provider cleanup scoping.
 */
async function resolveResourceProvider(resource: ManifestResource): Promise<string> {
  if (resource.resource_type === 'instance') {
    const instance = queryOne<{ provider: string }>(
      'SELECT provider FROM instances WHERE id = ?', [resource.resource_id]
    );
    return instance?.provider ?? 'unknown';
  }

  if (resource.resource_type === 'allocation') {
    const alloc = queryOne<{ instance_id: number }>(
      'SELECT instance_id FROM allocations WHERE id = ?', [resource.resource_id]
    );
    if (alloc) {
      const instance = queryOne<{ provider: string }>(
        'SELECT provider FROM instances WHERE id = ?', [alloc.instance_id]
      );
      return instance?.provider ?? 'unknown';
    }
    return 'unknown';
  }

  // Objects (logs, artifacts, snapshots, run_files) have a provider column
  const obj = queryOne<{ provider: string }>(
    'SELECT provider FROM objects WHERE id = ?', [resource.resource_id]
  );
  return obj?.provider ?? 'unknown';
}
```

> **Per-provider fan-out**: The cleanup nodes below operate on all resources
> regardless of provider. At L3/L4 implementation, each cleanup node MUST
> iterate over `resourcesByProvider` and process each provider group
> independently, so that a failure in one provider (e.g., AWS API timeout)
> does not block cleanup of another provider's resources (e.g., OrbStack).
> See L1 §PerProviderOrchestration and spec Ch04/Ch07.

### cleanup-allocations (Priority 90)

```typescript
/**
 * Clean allocation records with atomic transitions.
 *
 * CRITICAL: Uses atomic UPDATE with IN clause for all valid source statuses.
 * This prevents race conditions where allocation state changes between
 * status check and transition.
 *
 * Pattern: UPDATE ... SET status='COMPLETE' WHERE id=? AND status IN (...)
 * If 0 rows affected: allocation already COMPLETE or FAILED (idempotent)
 */
async function cleanupAllocations(
  ctx: WorkflowContext,
  input: CleanupManifestInput,
  deps: { 'load-resources': { resourcesByPriority: Map<number, ManifestResource[]> } }
): Promise<{ cleaned: number; errors: Array<{ resourceId: number; error: string }> }> {
  const { resourcesByPriority } = deps['load-resources'];
  const allocationResources = resourcesByPriority.get(90) || [];

  let cleaned = 0;
  const errors: Array<{ resourceId: number; error: string }> = [];

  for (const resource of allocationResources) {
    if (resource.resource_type !== 'allocation') continue;

    try {
      const result = await completeAllocationForCleanup(resource.resource_id);

      if (result.success) {
        cleaned++;
        logger.info('allocation_cleaned', {
          allocationId: resource.resource_id,
          manifestId: input.manifestId
        });
      } else {
        // NOT_FOUND or WRONG_STATE (already terminal) - skip silently
        logger.debug('allocation_cleanup_skipped', {
          allocationId: resource.resource_id,
          reason: result.reason,
          currentStatus: result.current?.status
        });
      }
    } catch (error) {
      errors.push({
        resourceId: resource.resource_id,
        error: (error as Error).message
      });
      logger.warn('allocation_cleanup_error', {
        allocationId: resource.resource_id,
        error: (error as Error).message
      });
      // Continue with other allocations (skip_and_continue policy)
    }
  }

  return { cleaned, errors };
}

/**
 * Complete allocation for cleanup using atomic transition.
 *
 * CRITICAL: Uses WHERE status IN (...) to accept any non-terminal state.
 * This is safe for cleanup because we're moving to terminal COMPLETE state.
 */
async function completeAllocationForCleanup(
  allocationId: number
): Promise<TransitionResult<Allocation>> {
  const db = getDatabase();

  return transaction(() => {
    // Phase 1: Read current state
    const allocation = queryOne<Allocation>(
      'SELECT * FROM allocations WHERE id = ?',
      [allocationId]
    );

    if (!allocation) {
      return transitionFailure<Allocation>('NOT_FOUND');
    }

    // Already in terminal state - idempotent success
    if (allocation.status === 'COMPLETE' || allocation.status === 'FAILED') {
      return transitionSuccess<Allocation>(allocation);
    }

    // Phase 2: Atomic UPDATE with IN clause for all non-terminal states
    // This prevents race where allocation transitions between our SELECT and UPDATE
    const now = Date.now();
    const result = db.prepare(`
      UPDATE allocations
      SET status = 'COMPLETE', completed_at = ?, updated_at = ?
      WHERE id = ? AND status IN ('AVAILABLE', 'CLAIMED', 'ACTIVE')
    `).run(now, now, allocationId);

    if (result.changes === 0) {
      // Race lost - status changed to terminal state between SELECT and UPDATE
      const latest = queryOne<Allocation>('SELECT * FROM allocations WHERE id = ?', [allocationId]);
      if (latest && (latest.status === 'COMPLETE' || latest.status === 'FAILED')) {
        // Terminal state is acceptable - idempotent
        return transitionSuccess<Allocation>(latest);
      }
      return transitionFailure<Allocation>('RACE_LOST', latest ?? undefined);
    }

    const updated = queryOne<Allocation>('SELECT * FROM allocations WHERE id = ?', [allocationId]);
    return transitionSuccess<Allocation>(updated!);
  });
}
```

### cleanup-runs (Priority 80)

```typescript
async function cleanupRuns(
  ctx: WorkflowContext,
  input: CleanupManifestInput,
  deps: { 'load-resources': { resourcesByPriority: Map<number, ManifestResource[]> } }
): Promise<{ cleaned: number; errors: Array<{ resourceId: number; error: string }> }> {
  const { resourcesByPriority } = deps['load-resources'];
  const runResources = resourcesByPriority.get(80) || [];

  let cleaned = 0;
  const errors: Array<{ resourceId: number; error: string }> = [];

  for (const resource of runResources) {
    if (resource.resource_type !== 'run') continue;

    try {
      const result = getDatabase().prepare(`
        UPDATE runs SET workflow_state = 'cleanup:complete'
        WHERE id = ? AND workflow_state != 'cleanup:complete'
      `).run(resource.resource_id);

      if (result.changes > 0) {
        cleaned++;
      }
    } catch (error) {
      errors.push({
        resourceId: resource.resource_id,
        error: (error as Error).message
      });
    }
  }

  return { cleaned, errors };
}
```

### cleanup-artifacts (Priority 70)

```typescript
/**
 * Clean artifact references with B21 double-processing protection.
 *
 * Uses cleanup_processed_at flag to prevent double-processing blob
 * reference decrements during cleanup (crash recovery safe).
 */
async function cleanupArtifacts(
  ctx: WorkflowContext,
  input: CleanupManifestInput,
  deps: { 'load-resources': { resourcesByPriority: Map<number, ManifestResource[]> } }
): Promise<{ cleaned: number; errors: Array<{ resourceId: number; error: string }> }> {
  const { resourcesByPriority } = deps['load-resources'];
  const artifactResources = resourcesByPriority.get(70) || [];

  let cleaned = 0;
  const errors: Array<{ resourceId: number; error: string }> = [];

  for (const resource of artifactResources) {
    if (resource.resource_type !== 'artifact') continue;

    try {
      // B21: Check if already processed
      const current = queryOne<{ cleanup_processed_at: number | null }>(
        'SELECT cleanup_processed_at FROM manifest_resources WHERE id = ?',
        [resource.id]
      );

      if (current?.cleanup_processed_at !== null) {
        // Already processed - skip
        continue;
      }

      // Process in transaction
      transaction(() => {
        const db = getDatabase();
        const now = Date.now();

        // Delete artifact object record
        db.prepare('DELETE FROM objects WHERE id = ?').run(resource.resource_id);

        // Mark as processed (B21)
        db.prepare(`
          UPDATE manifest_resources SET cleanup_processed_at = ?
          WHERE id = ?
        `).run(now, resource.id);
      });

      cleaned++;
    } catch (error) {
      errors.push({
        resourceId: resource.resource_id,
        error: (error as Error).message
      });
    }
  }

  return { cleaned, errors };
}
```

### cleanup-objects (Priority 60)

```typescript
/**
 * Clean generic objects with B21 double-processing protection.
 */
async function cleanupObjects(
  ctx: WorkflowContext,
  input: CleanupManifestInput,
  deps: { 'load-resources': { resourcesByPriority: Map<number, ManifestResource[]> } }
): Promise<{ cleaned: number; errors: Array<{ resourceId: number; error: string }> }> {
  const { resourcesByPriority } = deps['load-resources'];
  const objectResources = resourcesByPriority.get(60) || [];

  let cleaned = 0;
  const errors: Array<{ resourceId: number; error: string }> = [];

  for (const resource of objectResources) {
    if (!['log', 'run_file', 'snapshot'].includes(resource.resource_type)) continue;

    try {
      // B21: Check if already processed
      const current = queryOne<{ cleanup_processed_at: number | null }>(
        'SELECT cleanup_processed_at FROM manifest_resources WHERE id = ?',
        [resource.id]
      );

      if (current?.cleanup_processed_at !== null) {
        continue;
      }

      // S3 deletion: if the object references a blob with an S3 key,
      // delete the S3 data before removing the DB record. DB cascade
      // (ON DELETE RESTRICT) prevents blob row deletion while objects
      // reference it, but S3 is external and requires explicit API calls.
      const obj = queryOne<{ blob_id: number; provider: string }>(
        'SELECT blob_id, provider FROM objects WHERE id = ?',
        [resource.resource_id]
      );

      if (obj && obj.blob_id) {
        const blob = queryOne<{ s3_key: string | null }>(
          'SELECT s3_key FROM blobs WHERE id = ?',
          [obj.blob_id]
        );

        if (blob?.s3_key) {
          const client = getProviderClient(obj.provider);
          await client.deleteS3Object(blob.s3_key);
          logger.debug('s3_object_deleted', {
            blobId: obj.blob_id,
            s3Key: blob.s3_key,
            resourceId: resource.resource_id
          });
        }
      }

      transaction(() => {
        const db = getDatabase();
        const now = Date.now();

        db.prepare('DELETE FROM objects WHERE id = ?').run(resource.resource_id);
        db.prepare(`
          UPDATE manifest_resources SET cleanup_processed_at = ?
          WHERE id = ?
        `).run(now, resource.id);
      });

      cleaned++;
    } catch (error) {
      errors.push({
        resourceId: resource.resource_id,
        error: (error as Error).message
      });
    }
  }

  return { cleaned, errors };
}
```

### cleanup-instances (Priority 50)

```typescript
async function cleanupInstances(
  ctx: WorkflowContext,
  input: CleanupManifestInput,
  deps: { 'load-resources': { resourcesByPriority: Map<number, ManifestResource[]> } }
): Promise<{ cleaned: number; errors: Array<{ resourceId: number; error: string }> }> {
  const { resourcesByPriority } = deps['load-resources'];
  const instanceResources = resourcesByPriority.get(50) || [];

  let cleaned = 0;
  const errors: Array<{ resourceId: number; error: string }> = [];

  for (const resource of instanceResources) {
    if (resource.resource_type !== 'instance') continue;

    try {
      const instance = queryOne<{ id: number; provider: string; provider_id: string; workflow_state: string }>(
        'SELECT * FROM instances WHERE id = ?',
        [resource.resource_id]
      );

      if (!instance) continue;

      if (instance.workflow_state !== 'terminate:complete') {
        const client = getProviderClient(instance.provider);
        await client.terminate(instance.provider_id);

        getDatabase().prepare(`
          UPDATE instances SET workflow_state = 'terminate:complete'
          WHERE id = ?
        `).run(instance.id);
      }

      cleaned++;
    } catch (error) {
      errors.push({
        resourceId: resource.resource_id,
        error: (error as Error).message
      });
      // retry_then_skip policy - log for retry
      logger.error('instance_termination_failed', {
        instanceId: resource.resource_id,
        error: (error as Error).message
      });
    }
  }

  return { cleaned, errors };
}
```

### cleanup-snapshots (Priority 10)

```typescript
async function cleanupSnapshots(
  ctx: WorkflowContext,
  input: CleanupManifestInput,
  deps: { 'load-resources': { resourcesByPriority: Map<number, ManifestResource[]> } }
): Promise<{ cleaned: number; errors: Array<{ resourceId: number; error: string }> }> {
  const { resourcesByPriority } = deps['load-resources'];
  const snapshotResources = resourcesByPriority.get(10) || [];

  let cleaned = 0;
  const errors: Array<{ resourceId: number; error: string }> = [];

  for (const resource of snapshotResources) {
    if (resource.resource_type !== 'snapshot') continue;

    try {
      // Get snapshot details from objects table
      const snapshot = queryOne<{ metadata_json: string; provider: string }>(
        `SELECT o.metadata_json, o.provider
         FROM objects o
         WHERE o.id = ? AND o.type = 'snapshot'`,
        [resource.resource_id]
      );

      if (snapshot) {
        const metadata = JSON.parse(snapshot.metadata_json);
        const client = getProviderClient(snapshot.provider);
        await client.deleteSnapshot(metadata.provider_snapshot_id);
      }

      getDatabase().prepare('DELETE FROM objects WHERE id = ?').run(resource.resource_id);
      cleaned++;
    } catch (error) {
      errors.push({
        resourceId: resource.resource_id,
        error: (error as Error).message
      });
    }
  }

  return { cleaned, errors };
}
```

```typescript
async function deleteManifest(
  ctx: WorkflowContext,
  input: CleanupManifestInput
): Promise<{ deleted: boolean }> {
  const db = getDatabase();

  transaction(() => {
    // Cascade delete manifest_resources
    db.prepare('DELETE FROM manifest_resources WHERE manifest_id = ?').run(input.manifestId);
    db.prepare('DELETE FROM manifests WHERE id = ?').run(input.manifestId);
  });

  logger.info('manifest_deleted', { manifestId: input.manifestId });
  return { deleted: true };
}
```

### finalize

```typescript
async function finalize(
  ctx: WorkflowContext,
  input: CleanupManifestInput,
  deps: Record<string, { cleaned?: number; errors?: Array<{ resourceId: number; error: string }> }>
): Promise<CleanupManifestOutput> {
  // Aggregate metrics from all cleanup nodes
  let totalCleaned = 0;
  const allErrors: Array<{ resourceType: string; resourceId: number; error: string }> = [];

  for (const [nodeName, result] of Object.entries(deps)) {
    if (result.cleaned) {
      totalCleaned += result.cleaned;
    }
    if (result.errors) {
      for (const error of result.errors) {
        allErrors.push({
          resourceType: nodeName.replace('cleanup-', ''),
          ...error
        });
      }
    }
  }

  logger.info('cleanup_manifest_complete', {
    manifestId: input.manifestId,
    resourcesCleaned: totalCleaned,
    errorCount: allErrors.length
  });

  return {
    workflowId: ctx.workflowId,
    manifestId: input.manifestId,
    resourcesCleaned: totalCleaned,
    errors: allErrors,
    completedAt: Date.now()
  };
}
```

## CompensationForPartialCleanup

```typescript
/**
 * Track cleanup progress for crash recovery.
 *
 * On restart, resume from last_completed_priority + 1,
 * skipping resources at same priority with id <= last_completed_resource_id.
 */
interface ManifestCleanupState {
  manifest_id: number;
  last_completed_priority: number;
  last_completed_resource_id: number;
  started_at: number;
  updated_at: number;
}

function updateCleanupProgress(
  manifestId: number,
  priority: number,
  resourceId: number
): void {
  const now = Date.now();
  getDatabase().prepare(`
    INSERT INTO manifest_cleanup_state (manifest_id, last_completed_priority, last_completed_resource_id, started_at, updated_at)
    VALUES (?, ?, ?, ?, ?)
    ON CONFLICT (manifest_id) DO UPDATE SET
      last_completed_priority = excluded.last_completed_priority,
      last_completed_resource_id = excluded.last_completed_resource_id,
      updated_at = excluded.updated_at
  `).run(manifestId, priority, resourceId, now, now);
}
```
