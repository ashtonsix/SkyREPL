## NodeExecutor

```typescript
export const createAllocationExecutor: NodeExecutor<CreateAllocationInput, CreateAllocationOutput> = {
  name: 'create-allocation',
  idempotent: true,  // INSERT is guarded by (instance_id, run_id) uniqueness check

  async execute(ctx: NodeContext): Promise<CreateAllocationOutput> {
    const input = ctx.input as CreateAllocationInput;
    const db = getDatabase();

    // ── Input validation ──────────────────────────────────────────────
    if (!input.instanceId) {
      throw { code: 'VALIDATION_ERROR', message: 'instanceId is required' };
    }
    if (!input.runId) {
      throw { code: 'VALIDATION_ERROR', message: 'runId is required' };
    }

    // ── Idempotency check ─────────────────────────────────────────────
    // If an allocation already exists for this (instance_id, run_id), return it.
    // This handles crash-recovery re-execution.
    const existing = queryOne<{ id: number; instance_id: number; workdir: string }>(
      `SELECT id, instance_id, workdir FROM allocations
       WHERE instance_id = ? AND run_id = ? AND status IN ('CLAIMED', 'ACTIVE')`,
      [input.instanceId, input.runId]
    );

    if (existing) {
      ctx.log('info', 'Allocation already exists (idempotent)', {
        allocationId: existing.id,
      });
      ctx.emitResource('allocation', existing.id, 90);
      ctx.emitResource('run', input.runId, 80);
      return {
        allocationId: existing.id,
        instanceId: existing.instance_id,
        workdir: existing.workdir,
      };
    }

    // ── Determine workdir ─────────────────────────────────────────────
    const workdir = input.workdir ?? `/home/user/run-${input.runId}`;

    // ── Create allocation record ──────────────────────────────────────
    // Status starts as CLAIMED (run is bound but not yet executing).
    // Transition to ACTIVE happens in start-run after sync_complete.
    const now = Date.now();

    const result = db.prepare(`
      INSERT INTO allocations (
        instance_id, run_id, status, workdir,
        current_manifest_id, created_at, updated_at
      ) VALUES (?, ?, 'CLAIMED', ?, ?, ?, ?)
    `).run(
      input.instanceId,
      input.runId,
      workdir,
      input.manifestId,
      now,
      now
    );

    const allocationId = result.lastInsertRowid as number;

    // ── Emit resources to manifest ────────────────────────────────────
    // Allocation at priority 90 (cleaned up first: releases instance slots)
    ctx.emitResource('allocation', allocationId, 90);
    // Run at priority 80 (cleaned up after allocation)
    ctx.emitResource('run', input.runId, 80);

    ctx.log('info', 'Allocation created', {
      allocationId,
      instanceId: input.instanceId,
      runId: input.runId,
      workdir,
    });

    return {
      allocationId,
      instanceId: input.instanceId,
      workdir,
    };
  },

  // ── Compensation ──────────────────────────────────────────────────
  // Delete the allocation record. Instance cleanup is handled separately
  // by spawn-instance compensation.
  async compensate(ctx: NodeContext): Promise<void> {
    const output = ctx.output as CreateAllocationOutput | undefined;
    if (!output) return;

    ctx.log('warn', 'Compensating create-allocation', {
      allocationId: output.allocationId,
    });

    const db = getDatabase();
    db.prepare('DELETE FROM allocations WHERE id = ? AND status = ?')
      .run(output.allocationId, 'CLAIMED');
  },
};
```
