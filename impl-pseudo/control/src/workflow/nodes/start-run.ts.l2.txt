export interface StartRunInput {
  runId: number;

  allocationId: number;

  env?: Record<string, string>; // Environment variables

export interface StartRunOutput {

## NodeExecutor

```typescript
export const startRunExecutor: NodeExecutor<StartRunInput, StartRunOutput> = {
  name: 'start-run',
  idempotent: false,  // Sends command to agent: NOT safe to blindly retry
                      // (agent may already be executing). Engine will fail
                      // on crash-recovery rather than retry.

  async execute(ctx: NodeContext): Promise<StartRunOutput> {
    const input = ctx.input as StartRunInput;
    const db = getDatabase();

    // ── Input validation ──────────────────────────────────────────────
    if (!input.runId) {
      throw { code: 'VALIDATION_ERROR', message: 'runId is required' };
    }
    if (!input.instanceId) {
      throw { code: 'VALIDATION_ERROR', message: 'instanceId is required' };
    }
    if (!input.command) {
      throw { code: 'VALIDATION_ERROR', message: 'command is required' };
    }

    ctx.log('info', 'Starting run on instance', {
      runId: input.runId,
      instanceId: input.instanceId,
      allocationId: input.allocationId,
      fileCount: input.files.length,
    });

    // ── Step 1: Verify instance is connected ──────────────────────────
    // Instance must have an active SSE connection for command delivery.
    // If not connected, the command will be queued (SSEManager handles this).
    const instance = queryOne<{
      id: number;
      ip: string;
      workflow_state: string;
    }>(
      'SELECT id, ip, workflow_state FROM instances WHERE id = ?',
      [input.instanceId]
    );

    if (!instance) {
      throw {
        code: 'VALIDATION_ERROR',
        message: `Instance ${input.instanceId} not found`,
      };
    }

    // ── Step 2: Generate presigned download URLs for run files ─────────
    // Each file is stored as a content-addressable blob. The agent downloads
    // files from these presigned URLs during sync.
    const fileUrls = await generatePresignedUrls(
      input.files.map(f => f.checksum),
      'GET',
      TIMING.PRESIGNED_URL_TTL_MS ?? 3600_000  // 1 hour default
    );

    const filesWithUrls = input.files.map((file, i) => ({
      path: file.path,
      checksum: file.checksum,
      sizeBytes: file.sizeBytes,
      downloadUrl: fileUrls[i],
    }));

    // ── Step 3: Send start_run command to agent via SSE ────────────────
    // Single atomic message: agent receives files list + command together.
    // Agent downloads files, then starts the run process.
    // On completion of file download, agent sends sync_complete via log endpoint.
    const commandSent = await sseManager.sendCommand(String(input.instanceId), {
      type: 'start_run',
      run_id: input.runId,
      allocation_id: input.allocationId,
      command: input.command,
      workdir: input.workdir,
      env: input.env ?? {},
      files: filesWithUrls,
      artifact_patterns: input.artifactPatterns ?? [],
      max_duration_ms: input.maxDurationMs,
    });

    if (!commandSent) {
      ctx.log('warn', 'Agent not connected, start_run command queued', {
        instanceId: input.instanceId,
      });
      // Command is queued by SSEManager; agent will receive it on reconnect.
      // We still proceed to wait for sync_complete.
    }

    // ── Step 4: Wait for sync_complete event from agent ───────────────
    // Agent reports sync_complete via POST /v1/agent/logs with
    // stream="sync_complete". Control plane receives this and resolves
    // the waiting promise.
    //
    // Timeout: configurable, default 10 minutes for file sync.
    const syncTimeout = TIMING.SYNC_TIMEOUT_MS ?? 600_000; // 10 min

    const syncResult = await waitForAgentEvent(input.instanceId, 'sync_complete', {
      runId: input.runId,
      timeout: syncTimeout,
    });

    if (!syncResult.success) {
      // Sync failed or timed out
      ctx.log('error', 'File sync failed', {
        runId: input.runId,
        error: syncResult.error,
        timeout: syncResult.timeout,
      });

      // Transition allocation to FAILED
      db.prepare(`
        UPDATE allocations SET status = 'FAILED', updated_at = ?
        WHERE id = ? AND status = 'CLAIMED'
      `).run(Date.now(), input.allocationId);

      if (syncResult.timeout) {
        throw {
          code: 'OPERATION_TIMEOUT',
          message: `File sync timed out after ${syncTimeout}ms`,
        };
      }
      throw {
        code: 'PROVIDER_API_ERROR',
        message: `File sync failed: ${syncResult.error}`,
      };
    }

    // ── Step 5: Transition allocation CLAIMED -> ACTIVE ───────────────
    // Control plane owns this transition (not the agent).
    // Agent only reports events; state machine is server-side.
    const transitionResult = await activateAllocation(input.allocationId);

    if (!transitionResult.success) {
      ctx.log('error', 'Allocation transition CLAIMED->ACTIVE failed', {
        allocationId: input.allocationId,
        reason: transitionResult.reason,
        currentStatus: transitionResult.current?.status,
      });

      throw {
        code: 'INVALID_STATE_TRANSITION',
        message: `Allocation ${input.allocationId} transition CLAIMED->ACTIVE failed: ${transitionResult.reason}`,
      };
    }

    // ── Step 6: Update run record ─────────────────────────────────────
    const now = Date.now();
    db.prepare(`
      UPDATE runs SET status = 'running', started_at = ?
      WHERE id = ?
    `).run(now, input.runId);

    ctx.log('info', 'Run started, allocation ACTIVE', {
      runId: input.runId,
      allocationId: input.allocationId,
      filesSynced: input.files.length,
    });

    return {
      started: true,
      syncedAt: now,
      filesSynced: input.files.length,
    };
  },

  // ── Compensation ──────────────────────────────────────────────────
  // Best-effort: send cancel_run to agent, update run status.
  // Allocation state is NOT reverted (cleanup-manifest handles that).
  async compensate(ctx: NodeContext): Promise<void> {
    const input = ctx.input as StartRunInput;

    ctx.log('warn', 'Compensating start-run', { runId: input.runId });

    // Send cancel command to agent (fire-and-forget)
    await sseManager.sendCommand(String(input.instanceId), {
      type: 'cancel_run',
      run_id: input.runId,
    });

    // Update run status
    const db = getDatabase();
    db.prepare(`
      UPDATE runs SET status = 'cancelled', updated_at = ?
      WHERE id = ?
    `).run(Date.now(), input.runId);
  },
};
```

## EventWaiting

```typescript
/**
 * Wait for a specific agent event by registering a promise that resolves
 * when the event is received via the agent log endpoint.
 *
 * The agent sends special log entries (stream="sync_complete", stream="run_complete")
 * that are detected by the log ingestion handler. When detected, the handler
 * resolves the corresponding registered promise.
 *
 * Implementation: Map<string, { resolve, reject }> keyed by "${instanceId}:${eventType}:${runId}"
 */
async function waitForAgentEvent(
  instanceId: number,
  eventType: string,
  options: { runId: number; timeout: number }
): Promise<{ success: boolean; timeout?: boolean; error?: string; data?: any }> {
  return new Promise((resolve) => {
    const key = `${instanceId}:${eventType}:${options.runId}`;

    // Register callback in global event registry
    const timer = setTimeout(() => {
      agentEventRegistry.delete(key);
      resolve({ success: false, timeout: true });
    }, options.timeout);

    agentEventRegistry.set(key, (data: any) => {
      clearTimeout(timer);
      agentEventRegistry.delete(key);

      if (data.success === false) {
        resolve({ success: false, error: data.error });
      } else {
        resolve({ success: true, data });
      }
    });
  });
}

// Global registry: log ingestion handler calls these when special events arrive.
// See agent.ts POST /v1/agent/logs handler for the dispatch logic.
const agentEventRegistry = new Map<string, (data: any) => void>();
```

## ErrorHandling

| Error                      | Source          | Response                                     |
|----------------------------|----------------|----------------------------------------------|
| VALIDATION_ERROR           | Input          | Fail immediately                             |
| PROVIDER_API_ERROR         | SSE send       | Retry with backoff                           |
| OPERATION_TIMEOUT          | Sync timeout   | Compensate + fail                            |
| INVALID_STATE_TRANSITION   | Allocation atomic transition | Fail immediately (state inconsistency)       |
| Agent sync failure         | Agent event    | Allocation -> FAILED, compensate + fail      |
