export interface FinalizeInput {

## NodeExecutor

```typescript
export const finalizeExecutor: NodeExecutor<FinalizeInput, FinalizeOutput> = {
  name: 'finalize',
  idempotent: true,  // Transitions are atomic: re-execution is safe

  async execute(ctx: NodeContext): Promise<FinalizeOutput> {
    const input = ctx.input as FinalizeInput;
    const db = getDatabase();

    // ── Input validation ──────────────────────────────────────────────
    if (!input.allocationId) {
      throw { code: 'VALIDATION_ERROR', message: 'allocationId is required' };
    }
    if (!input.manifestId) {
      throw { code: 'VALIDATION_ERROR', message: 'manifestId is required' };
    }

    ctx.log('info', 'Finalizing run', {
      runId: input.runId,
      allocationId: input.allocationId,
      exitCode: input.exitCode,
      spotInterrupted: input.spotInterrupted,
    });

    // ── Step 1: Transition allocation ACTIVE -> COMPLETE ──────────────
    // Uses atomic transition from state-transitions.ts. If allocation is already FAILED
    // (e.g., heartbeat timeout race), completeAllocation treats it as success.
    const now = Date.now();

    const debugHoldUntil = (input.holdDurationMs && input.holdDurationMs > 0)
      ? now + input.holdDurationMs
      : null;

    const allocationResult = await completeAllocation(input.allocationId, {
      debugHoldUntil: debugHoldUntil ?? undefined,
    });

    let allocationStatus: string;
    if (allocationResult.success) {
      allocationStatus = allocationResult.data.status;
      ctx.log('info', 'Allocation transitioned to COMPLETE', {
        allocationId: input.allocationId,
        debugHoldUntil,
      });
    } else {
      // completeAllocation handles FAILED-already case internally.
      // If it still fails, log and continue (finalize is skip_and_continue).
      allocationStatus = allocationResult.current?.status ?? 'unknown';
      ctx.log('warn', 'Allocation transition to COMPLETE failed', {
        allocationId: input.allocationId,
        reason: allocationResult.reason,
        currentStatus: allocationStatus,
      });
    }

    // ── Step 2: Update run status ─────────────────────────────────────
    const runStatus = input.exitCode === 0 ? 'completed' : 'failed';
    db.prepare(`
      UPDATE runs SET status = ?, updated_at = ?
      WHERE id = ?
    `).run(runStatus, now, input.runId);

    ctx.log('info', 'Run status updated', {
      runId: input.runId,
      status: runStatus,
      exitCode: input.exitCode,
    });

    // ── Step 3: Clean up SSE command queue for this instance ──────────
    // Remove any pending commands for this instance that belong to this run.
    // Close CLI log subscribers for this run.
    sseManager.closeRunSubscribers(String(input.runId), {
      type: 'status',
      status: runStatus === 'completed' ? 'completed' : 'failed',
      exit_code: input.exitCode,
    });

    // ── Step 4: Conditional snapshot (CB pattern) ─────────────────────
    // Only create snapshot if requested AND run succeeded (exitCode === 0).
    // Snapshot creation is a subworkflow (create-snapshot intent).
    let snapshotId: number | undefined;

    if (input.createSnapshot && input.exitCode === 0 && !input.spotInterrupted) {
      ctx.log('info', 'Creating post-run snapshot', {
        instanceId: input.instanceId,
      });

      try {
        const snapshotWorkflow = await ctx.spawnSubworkflow('create-snapshot', {
          instanceId: input.instanceId,
          runId: input.runId,
          name: `run-${input.runId}-snapshot-${now}`,
          initChecksum: input.initChecksum,
        });

        const snapshotResult = await snapshotWorkflow.wait();

        if (snapshotResult.status === 'completed') {
          snapshotId = snapshotResult.output?.snapshotId as number;
          ctx.emitResource('object', snapshotId, 10); // Priority 10: last to clean up
          ctx.log('info', 'Snapshot created', { snapshotId });
        } else {
          // Snapshot failure does not fail the finalize node
          ctx.log('warn', 'Snapshot creation failed, continuing', {
            error: snapshotResult.error,
          });
        }
      } catch (error) {
        // Snapshot failure is non-fatal for finalize
        ctx.log('warn', 'Snapshot subworkflow error, continuing', { error });
      }
    }

    // ── Step 5: Seal manifest ─────────────────────────────────────────
    // DRAFT -> SEALED transition. Sets expires_at for cleanup scheduling.
    // All resources are now locked; no more emissions allowed.
    const expiresAt = now + (TIMING.MANIFEST_EXPIRY_MS ?? 86_400_000); // Default 24h

    const sealResult = await sealManifest(input.manifestId, expiresAt);

    let manifestSealed = false;
    if (sealResult.success) {
      manifestSealed = true;
      ctx.log('info', 'Manifest sealed', {
        manifestId: input.manifestId,
        expiresAt,
      });
    } else {
      // Manifest may already be sealed (idempotent) or in unexpected state
      ctx.log('warn', 'Manifest seal failed', {
        manifestId: input.manifestId,
        reason: sealResult.reason,
      });
      // Check if already sealed
      const manifest = queryOne<{ status: string }>(
        'SELECT status FROM manifests WHERE id = ?',
        [input.manifestId]
      );
      if (manifest?.status === 'SEALED') {
        manifestSealed = true;
      }
    }

    ctx.log('info', 'Finalize complete', {
      runId: input.runId,
      allocationStatus,
      runStatus,
      manifestSealed,
      snapshotId,
    });

    return {
      allocationStatus,
      runStatus,
      manifestSealed,
      snapshotId,
    };
  },

  // ── Compensation ──────────────────────────────────────────────────
  // Finalize is the terminal node: compensation is rarely needed.
  // If called, attempt to unseal manifest (allow further modifications).
  async compensate(ctx: NodeContext): Promise<void> {
    const input = ctx.input as FinalizeInput;

    ctx.log('warn', 'Compensating finalize (rare)', {
      manifestId: input.manifestId,
    });

    // Unseal manifest back to DRAFT (only if still SEALED)
    const db = getDatabase();
    db.prepare(`
      UPDATE manifests SET status = 'DRAFT', expires_at = NULL, updated_at = ?
      WHERE id = ? AND status = 'SEALED'
    `).run(Date.now(), input.manifestId);

    // Note: snapshot cleanup is handled by cleanup-manifest intent
    // if the snapshot was emitted to the manifest.
  },
};
```

## ErrorHandling

| Error              | Source           | Response                                 |
|--------------------|------------------|------------------------------------------|
| VALIDATION_ERROR   | Input            | Fail immediately                         |
| Allocation atomic  | State transition | Log warning, continue (skip_and_continue)|
| Snapshot failure   | Subworkflow      | Log warning, continue (non-fatal)        |
| Manifest seal fail | State transition | Log warning, check if already sealed     |
| DATABASE_ERROR     | DB               | Fail (but finalize has skip_and_continue) |

Per the launch-run error handling table (Ch7 §7.1), finalize-run uses
`skip_and_continue` error handling: failures in this node do not fail the
overall workflow.
