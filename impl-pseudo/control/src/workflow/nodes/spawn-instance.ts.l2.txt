```typescript
export interface SpawnInstanceInput {
  runId: number;
  manifestId: number;
  spec: string;            // Instance spec (e.g., "gpu-a100-80gb")
  provider: string;        // Provider name
  region?: string;         // Target region
  spot?: boolean;          // Use spot/preemptible instances
  initChecksum?: string;   // Snapshot checksum for image-based boot
  bootstrap: BootstrapConfig;  // Agent bootstrap configuration
}

export interface SpawnInstanceOutput {
  instanceId: number;      // Our DB record ID
  providerId: string;      // Provider's instance identifier
  ip: string | null;       // IP address (may be null initially)
  status: string;          // Workflow state after spawn
}

interface BootstrapConfig {
  controlPlaneUrl: string;
  authToken: string;
  agentVersion: string;
}
```

## NodeExecutor

```typescript
export const spawnInstanceExecutor: NodeExecutor<SpawnInstanceInput, SpawnInstanceOutput> = {
  name: 'spawn-instance',
  idempotent: true,  // Uses idempotency key: safe to retry after crash

  async execute(ctx: NodeContext): Promise<SpawnInstanceOutput> {
    const input = ctx.input as SpawnInstanceInput;
    const db = getDatabase();

    // ── Input validation ──────────────────────────────────────────────
    if (!input.spec) {
      throw { code: 'VALIDATION_ERROR', message: 'spec is required' };
    }
    if (!input.provider) {
      throw { code: 'VALIDATION_ERROR', message: 'provider is required' };
    }

    ctx.log('info', 'Spawning new instance', {
      spec: input.spec,
      provider: input.provider,
      region: input.region,
      spot: input.spot,
    });

    // ── Step 1: Generate idempotency key ──────────────────────────────
    // Ensures crash-recovery re-execution finds existing instance record
    const spawnKey = `spawn:${input.manifestId}:${input.runId}:${input.spec}:${input.initChecksum ?? 'none'}`;

    // ── Step 2: Check for existing instance (idempotent) ──────────────
    const existing = queryOne<{
      id: number;
      provider_id: string;
      ip: string;
      workflow_state: string;
    }>(
      `SELECT id, provider_id, ip, workflow_state FROM instances
       WHERE spawn_key = ? AND workflow_state NOT IN ('spawn:error', 'terminated')`,
      [spawnKey]
    );

    if (existing) {
      ctx.log('info', 'Found existing instance from previous attempt', {
        instanceId: existing.id,
        status: existing.workflow_state,
      });

      // If provider_id is set, instance was already spawned at the provider
      if (existing.provider_id) {
        return {
          instanceId: existing.id,
          providerId: existing.provider_id,
          ip: existing.ip,
          status: existing.workflow_state,
        };
      }

      // provider_id is null: Phase 1 completed but Phase 2 didn't.
      // Fall through to Phase 2 with existing DB record.
    }

    // ── Step 3: Phase 1 - Reserve instance record in DB ───────────────
    // DB record created BEFORE provider call. If we crash after this but
    // before the provider call, orphan detection will find the record
    // (status='spawn:pending', no provider_id).
    let instanceId: number;

    if (existing) {
      instanceId = existing.id;
    } else {
      const now = Date.now();
      const authTokenHash = generateAuthTokenHash();

      const result = db.prepare(`
        INSERT INTO instances (
          provider, region, spec, workflow_state, provider_id, ip,
          spawn_key, is_spot, init_checksum, current_manifest_id,
          auth_token_hash, created_at, updated_at
        ) VALUES (?, ?, ?, 'spawn:pending', NULL, NULL, ?, ?, ?, ?, ?, ?, ?)
      `).run(
        input.provider,
        input.region ?? null,
        input.spec,
        spawnKey,
        input.spot ? 1 : 0,
        input.initChecksum ?? null,
        input.manifestId,
        authTokenHash,
        now,
        now
      );

      instanceId = result.lastInsertRowid as number;
    }

    // Emit instance to manifest (priority 50: cleaned up after allocations/runs)
    ctx.emitResource('instance', instanceId, 50);

    // ── Step 4: Phase 2 - Call provider.spawn() ───────────────────────
    // This is the external side effect. If it fails, compensation will
    // terminate the provider instance if a provider_id was returned.
    const client = getProviderClient(input.provider);

    let snapshotId: string | undefined;
    if (input.initChecksum) {
      snapshotId = await findProviderSnapshotId(input.provider, input.initChecksum, input.spec);
    }

    const bootstrap = client.generateBootstrap({
      controlPlaneUrl: input.bootstrap.controlPlaneUrl,
      instanceId: String(instanceId),
      authToken: input.bootstrap.authToken,
      agentVersion: input.bootstrap.agentVersion,
    });

    let providerResult: { providerId: string; ip: string | null; status: string };
    try {
      providerResult = await client.spawn({
        spec: input.spec,
        region: input.region,
        spot: input.spot ?? false,
        snapshotId,
        bootstrap,
        tags: {
          'skyrepl:manifest_id': String(input.manifestId),
          'skyrepl:run_id': String(input.runId),
          'skyrepl:instance_id': String(instanceId),
          'skyrepl:spawn_key': spawnKey,
          'Name': `repl-${input.manifestId}-${instanceId}`,
        },
      });
    } catch (error: any) {
      // Update DB to reflect spawn failure
      db.prepare(`
        UPDATE instances SET workflow_state = 'spawn:error', updated_at = ?
        WHERE id = ?
      `).run(Date.now(), instanceId);

      // Re-throw with structured error code for engine retry logic
      if (error.code === 'CAPACITY_UNAVAILABLE') {
        throw {
          code: 'CAPACITY_UNAVAILABLE',
          message: `No capacity in ${input.region ?? 'default region'}`,
          details: { alternative_regions: error.alternativeRegions },
        };
      }
      if (error.code === 'RATE_LIMITED') {
        throw {
          code: 'RATE_LIMITED',
          message: 'Provider rate limit exceeded',
          details: { retry_after_ms: error.retryAfterMs },
        };
      }
      throw {
        code: 'PROVIDER_API_ERROR',
        message: `Provider spawn failed: ${error.message}`,
      };
    }

    // ── Step 5: Poll until instance is running ────────────────────────
    const pollInterval = TIMING.BOOT_POLL_INTERVAL_MS ?? 5000;
    const bootTimeout = TIMING.INSTANCE_BOOT_TIMEOUT_MS ?? 300_000; // 5 min
    const deadline = Date.now() + bootTimeout;

    while (Date.now() < deadline) {
      ctx.checkCancellation();

      const instanceStatus = await client.getInstanceStatus(providerResult.providerId);

      if (instanceStatus.state === 'running') {
        providerResult.ip = instanceStatus.ip ?? providerResult.ip;
        break;
      }
      if (instanceStatus.state === 'error' || instanceStatus.state === 'terminated') {
        db.prepare(`
          UPDATE instances SET workflow_state = 'spawn:error', provider_id = ?, updated_at = ?
          WHERE id = ?
        `).run(providerResult.providerId, Date.now(), instanceId);

        throw {
          code: 'PROVIDER_API_ERROR',
          message: `Instance entered ${instanceStatus.state} state during boot`,
        };
      }

      await ctx.sleep(pollInterval);
    }

    if (Date.now() >= deadline) {
      db.prepare(`
        UPDATE instances SET workflow_state = 'spawn:error', provider_id = ?, updated_at = ?
        WHERE id = ?
      `).run(providerResult.providerId, Date.now(), instanceId);

      throw {
        code: 'OPERATION_TIMEOUT',
        message: `Instance boot timed out after ${bootTimeout}ms`,
      };
    }

    // ── Step 6: Update DB with provider details ───────────────────────
    const now = Date.now();
    db.prepare(`
      UPDATE instances
      SET provider_id = ?, ip = ?, workflow_state = 'launch-run:provisioning',
          last_heartbeat = ?, updated_at = ?
      WHERE id = ?
    `).run(providerResult.providerId, providerResult.ip, now, now, instanceId);

    ctx.log('info', 'Instance spawned and running', {
      instanceId,
      providerId: providerResult.providerId,
      ip: providerResult.ip,
    });

    // ── Step 7: Return output ─────────────────────────────────────────
    return {
      instanceId,
      providerId: providerResult.providerId,
      ip: providerResult.ip,
      status: 'launch-run:provisioning',
    };
  },

  // ── Compensation ──────────────────────────────────────────────────
  // Called if this node or a downstream node fails.
  // Terminates the provider instance if one was created.
  async compensate(ctx: NodeContext): Promise<void> {
    const output = ctx.output as SpawnInstanceOutput | undefined;
    if (!output) return; // Node never produced output, nothing to compensate

    ctx.log('warn', 'Compensating spawn-instance', { instanceId: output.instanceId });

    const db = getDatabase();
    const instance = queryOne<{ id: number; provider_id: string | null; provider: string }>(
      'SELECT id, provider_id, provider FROM instances WHERE id = ?',
      [output.instanceId]
    );

    if (!instance) return; // Instance record not found, nothing to do

    // If provider_id exists, terminate at provider level
    if (instance.provider_id) {
      try {
        const client = getProviderClient(instance.provider);
        await client.terminate(instance.provider_id);
        ctx.log('info', 'Provider instance terminated during compensation', {
          instanceId: output.instanceId,
          providerId: instance.provider_id,
        });
      } catch (error) {
        // Best-effort: log and continue. Orphan scan will catch this.
        ctx.log('error', 'Failed to terminate provider instance during compensation', {
          instanceId: output.instanceId,
          error,
        });
      }
    }

    // Mark instance as compensated in DB
    db.prepare(`
      UPDATE instances SET workflow_state = 'launch-run:compensated', updated_at = ?
      WHERE id = ?
    `).run(Date.now(), output.instanceId);
  },
};
```

```typescript
/**
 * Find a provider-side snapshot ID matching the requested spec and checksum.
 * Returns the provider's snapshot identifier (e.g., AWS AMI ID) or undefined.
 */
async function findProviderSnapshotId(
  provider: string,
  initChecksum: string,
  spec: string
): Promise<string | undefined> {
  const snapshot = queryOne<{ provider_object_id: string }>(
    `SELECT o.provider_object_id FROM objects o
     WHERE o.type = 'snapshot'
       AND o.provider = ?
       AND json_extract(o.metadata_json, '$.init_checksum') = ?
       AND json_extract(o.metadata_json, '$.spec') = ?
     ORDER BY o.created_at DESC
     LIMIT 1`,
    [provider, initChecksum, spec]
  );

  return snapshot?.provider_object_id;
}
```

## ErrorHandling

| Error                  | Source         | Response                                          |
|------------------------|----------------|---------------------------------------------------|
| VALIDATION_ERROR       | Input          | Fail immediately (missing spec/provider)          |
| CAPACITY_UNAVAILABLE   | Provider       | Retry with alternative region (engine RWA)        |
| RATE_LIMITED           | Provider       | Retry same params with backoff (engine retry)     |
| PROVIDER_API_ERROR     | Provider       | Retry with backoff, max 3 attempts                |
| OPERATION_TIMEOUT      | Boot poll      | Retry with backoff, max 2 attempts                |
| DATABASE_ERROR         | DB             | Fail immediately (non-retryable)                  |
