## NodeExecutor

```typescript
export const claimAllocationExecutor: NodeExecutor<ClaimAllocationInput, ClaimAllocationOutput> = {
  name: 'claim-allocation',
  idempotent: true,  // CAS-based: re-execution is safe (already CLAIMED = success)

  async execute(ctx: NodeContext): Promise<ClaimAllocationOutput> {
    const input = ctx.input as ClaimAllocationInput;
    const db = getDatabase();

    // ── Input validation ──────────────────────────────────────────────
    if (!input.allocationId) {
      throw { code: 'VALIDATION_ERROR', message: 'allocationId is required' };
    }
    if (!input.runId) {
      throw { code: 'VALIDATION_ERROR', message: 'runId is required' };
    }

    ctx.log('info', 'Claiming warm pool allocation', {
      allocationId: input.allocationId,
      runId: input.runId,
    });

    // ── Pre-claim validation: single-tenant check ─────────────────────
    // Verify no other ACTIVE allocation exists on the target instance.
    // Enforces single-tenant model (one active run per instance at a time).
    const allocation = queryOne<{ instance_id: number }>(
      'SELECT instance_id FROM allocations WHERE id = ?',
      [input.allocationId]
    );

    if (!allocation) {
      throw {
        code: 'VALIDATION_ERROR',
        message: `Allocation ${input.allocationId} not found`,
      };
    }

    const activeCount = queryOne<{ count: number }>(
      `SELECT COUNT(*) as count FROM allocations
       WHERE instance_id = ? AND status = 'ACTIVE'`,
      [allocation.instance_id]
    );

    if (activeCount && activeCount.count > 0) {
      throw {
        code: 'INVALID_STATE_TRANSITION',
        message: `Instance ${allocation.instance_id} already has an active allocation`,
      };
    }

    // ── Atomic CAS claim: AVAILABLE -> CLAIMED ────────────────────────
    // Uses two-phase CAS from state-transitions.ts.
    // If another workflow claimed this allocation between resolve-instance
    // and now, the CAS fails with WRONG_STATE or RACE_LOST.
    const result = await claimAllocation(input.allocationId, input.runId);

    if (!result.success) {
      // Claim failed: allocation was taken by another workflow or changed state.
      // The CB triggerOnError pattern at this node will route to spawn-instance.
      ctx.log('warn', 'Warm pool claim failed', {
        allocationId: input.allocationId,
        reason: result.reason,
        currentStatus: result.current?.status,
      });

      throw {
        code: 'CAPACITY_UNAVAILABLE',
        message: `Warm pool claim failed: ${result.reason}`,
        details: {
          allocationId: input.allocationId,
          currentStatus: result.current?.status,
        },
      };
    }

    const claimed = result.data;

    // ── Update manifest association ───────────────────────────────────
    db.prepare(`
      UPDATE allocations SET current_manifest_id = ?, updated_at = ?
      WHERE id = ?
    `).run(input.manifestId, Date.now(), claimed.id);

    // ── Emit allocation to manifest ───────────────────────────────────
    ctx.emitResource('allocation', claimed.id, 90);

    ctx.log('info', 'Warm pool allocation claimed', {
      allocationId: claimed.id,
      instanceId: claimed.instance_id,
      runId: input.runId,
    });

    return {
      allocationId: claimed.id,
      instanceId: claimed.instance_id,
      fromWarmPool: true,
    };
  },

  // ── Compensation ──────────────────────────────────────────────────
  // Release the claimed allocation back to the warm pool.
  // Resets status from CLAIMED to AVAILABLE and clears run_id.
  async compensate(ctx: NodeContext): Promise<void> {
    const output = ctx.output as ClaimAllocationOutput | undefined;
    if (!output) return;

    ctx.log('warn', 'Compensating claim-allocation: releasing back to warm pool', {
      allocationId: output.allocationId,
    });

    const db = getDatabase();
    const now = Date.now();

    // Only release if still in CLAIMED state (don't clobber ACTIVE/COMPLETE)
    const result = db.prepare(`
      UPDATE allocations
      SET status = 'AVAILABLE', run_id = NULL, current_manifest_id = NULL, updated_at = ?
      WHERE id = ? AND status = 'CLAIMED'
    `).run(now, output.allocationId);

    if (result.changes > 0) {
      ctx.log('info', 'Allocation released back to warm pool', {
        allocationId: output.allocationId,
      });
    } else {
      ctx.log('info', 'Allocation not in CLAIMED state, skipping release', {
        allocationId: output.allocationId,
      });
    }
  },
};
```

## CBTriggerOnErrorIntegration

When this node throws (claim fails), the workflow engine's CB triggerOnError
pattern routes execution to spawn-instance as a fallback (Ch7 §7.1):

```
claim-allocation FAILS
  -> CB triggerOnError catches the error
  -> Routes to spawn-instance node (cold path)
  -> spawn-instance creates fresh instance
  -> Both paths converge at create-allocation
```

## ErrorHandling

| Error                      | Source      | Response                                        |
|----------------------------|------------|--------------------------------------------------|
| VALIDATION_ERROR           | Input      | Fail immediately (missing allocationId/runId)    |
| INVALID_STATE_TRANSITION   | Pre-check  | Fail immediately (active allocation on instance) |
| CAPACITY_UNAVAILABLE       | CAS claim  | CB triggerOnError routes to spawn-instance       |
| DATABASE_ERROR             | DB         | Fail immediately (non-retryable)                 |
