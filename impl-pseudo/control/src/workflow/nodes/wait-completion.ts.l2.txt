## NodeExecutor

```typescript
export const waitCompletionExecutor: NodeExecutor<WaitCompletionInput, WaitCompletionOutput> = {
  name: 'wait-completion',
  idempotent: false,  // Waiting is tied to live agent state: not safely retriable

  async execute(ctx: NodeContext): Promise<WaitCompletionOutput> {
    const input = ctx.input as WaitCompletionInput;
    const db = getDatabase();

    // ── Input validation ──────────────────────────────────────────────
    if (!input.runId) {
      throw { code: 'VALIDATION_ERROR', message: 'runId is required' };
    }
    if (!input.instanceId) {
      throw { code: 'VALIDATION_ERROR', message: 'instanceId is required' };
    }

    // ── Determine timeout ─────────────────────────────────────────────
    const maxDuration = input.maxDurationMs ?? TIMING.DEFAULT_RUN_TIMEOUT_MS ?? 86_400_000; // 24h

    ctx.log('info', 'Waiting for run completion', {
      runId: input.runId,
      instanceId: input.instanceId,
      maxDurationMs: maxDuration,
    });

    // ── Check if run already completed (crash recovery) ───────────────
    const existingRun = queryOne<{
      exit_code: number | null;
      started_at: number;
      finished_at: number | null;
      workflow_state: string;
    }>(
      'SELECT exit_code, started_at, finished_at, workflow_state FROM runs WHERE id = ?',
      [input.runId]
    );

    if (existingRun?.finished_at) {
      ctx.log('info', 'Run already completed (detected on recovery)', {
        runId: input.runId,
        exitCode: existingRun.exit_code,
      });
      return {
        exitCode: existingRun.exit_code ?? 1,
        completedAt: existingRun.finished_at,
        durationMs: existingRun.finished_at - existingRun.started_at,
        spotInterrupted: existingRun.workflow_state === 'run:spot_interrupted',
        timedOut: existingRun.workflow_state === 'run:timeout',
      };
    }

    // ── Wait for run_complete event from agent ────────────────────────
    // Agent sends run_complete via POST /v1/agent/logs when the run
    // process exits (or is killed by spot interruption / timeout).
    // Uses promise/callback registry (see start-run.ts.l2.txt §EventWaiting).
    const completionResult = await waitForAgentEvent(input.instanceId, 'run_complete', {
      runId: input.runId,
      timeout: maxDuration,
    });

    const now = Date.now();

    // ── Handle timeout ────────────────────────────────────────────────
    if (completionResult.timeout) {
      ctx.log('warn', 'Run timed out', {
        runId: input.runId,
        maxDurationMs: maxDuration,
      });

      // Send cancel command to agent (fire-and-forget via SSE)
      await sseManager.sendCommand(String(input.instanceId), {
        type: 'cancel_run',
        run_id: input.runId,
        reason: 'timeout',
      });

      // Update run record with timeout status
      db.prepare(`
        UPDATE runs SET exit_code = 124, workflow_state = 'run:timeout', finished_at = ?
        WHERE id = ?
      `).run(now, input.runId);

      const startedAt = existingRun?.started_at ?? now;

      return {
        exitCode: 124,   // Convention: 124 = timeout (matches GNU timeout)
        completedAt: now,
        durationMs: now - startedAt,
        spotInterrupted: false,
        timedOut: true,
      };
    }

    // ── Handle normal completion ──────────────────────────────────────
    const exitCode = completionResult.data?.exit_code ?? 1;
    const spotInterrupted = completionResult.data?.spot_interrupted ?? false;

    // Update run record
    db.prepare(`
      UPDATE runs SET exit_code = ?,
        workflow_state = ?,
        finished_at = ?
      WHERE id = ?
    `).run(
      exitCode,
      spotInterrupted ? 'run:spot_interrupted' : 'run:complete',
      now,
      input.runId
    );

    const startedAt = existingRun?.started_at ?? now;

    ctx.log('info', 'Run completed', {
      runId: input.runId,
      exitCode,
      spotInterrupted,
      durationMs: now - startedAt,
    });

    return {
      exitCode,
      completedAt: now,
      durationMs: now - startedAt,
      spotInterrupted,
      timedOut: false,
    };
  },

  // ── Compensation ──────────────────────────────────────────────────
  // Best-effort: send cancel_run to agent if run is still executing.
  // This is used when the workflow is cancelled or a downstream node fails.
  async compensate(ctx: NodeContext): Promise<void> {
    const input = ctx.input as WaitCompletionInput;

    ctx.log('warn', 'Compensating wait-completion', { runId: input.runId });

    // Check if run is still in progress
    const run = queryOne<{ finished_at: number | null }>(
      'SELECT finished_at FROM runs WHERE id = ?',
      [input.runId]
    );

    if (run && !run.finished_at) {
      // Run still executing: send cancel command
      await sseManager.sendCommand(String(input.instanceId), {
        type: 'cancel_run',
        run_id: input.runId,
        reason: 'workflow_compensation',
      });

      // Mark run as cancelled
      const db = getDatabase();
      db.prepare(`
        UPDATE runs SET status = 'cancelled', workflow_state = 'run:cancelled',
          finished_at = ?
        WHERE id = ? AND finished_at IS NULL
      `).run(Date.now(), input.runId);
    }
  },
};
```

## EventWaiting

Uses the same `waitForAgentEvent` mechanism as start-run.ts. The agent sends
`run_complete` via the log endpoint when:
- Run process exits normally (exit_code = process exit code)
- Run process killed by SIGTERM/SIGKILL (exit_code = 128 + signal)
- Spot interruption detected (spot_interrupted = true, exit_code from checkpoint)

See `start-run.ts.l2.txt` for the `waitForAgentEvent` implementation.

## ErrorHandling

| Error              | Source        | Response                                    |
|--------------------|---------------|---------------------------------------------|
| VALIDATION_ERROR   | Input         | Fail immediately                            |
| OPERATION_TIMEOUT  | Run timeout   | Send cancel_run, mark run as timeout        |
| DATABASE_ERROR     | DB            | Fail immediately (non-retryable)            |
