## ConnectionManagement

```typescript
let connection: Database | null = null;

export function initDatabase(path: string): Database {
  connection = new Database(path);

  // Enable WAL mode for better concurrency
  connection.pragma('journal_mode = WAL');

  // Enable foreign keys
  connection.pragma('foreign_keys = ON');

  // Busy timeout (milliseconds) - SQLite waits this long when table is locked
  connection.pragma('busy_timeout = 5000');

  return connection;
}

export function getDatabase(): Database {
  if (!connection) {
    throw DATABASE_CONNECTION_ERROR('Database not initialized');
  }
  if (!connection.open) {
    throw DATABASE_CONNECTION_ERROR('Database connection closed');
  }
  return connection;
}

export function closeDatabase(): void {
  if (connection) {
    connection.close();
    connection = null;
  }
}
```

## Migrations

```typescript
interface Migration {
  version: number;
  sql: string;
}

const MIGRATIONS: Migration[] = [
  // Loaded from migrations/*.sql files
];

export function getMigrationVersion(): number {
  const db = getDatabase();
  const result = db.prepare('SELECT version FROM schema_version').get() as { version: number } | undefined;
  return result?.version ?? 0;
}

function getLatestMigrationVersion(): number {
  return MIGRATIONS.length;
}

export function runMigrations(): void {
  const db = getDatabase();
  const current = getMigrationVersion();
  const target = getLatestMigrationVersion();

  if (current > target) {
    throw MIGRATION_ERROR(`Database version ${current} is newer than code version ${target}`);
  }

  for (const migration of MIGRATIONS.slice(current)) {
    try {
      db.transaction(() => {
        db.exec(migration.sql);
        db.prepare('UPDATE schema_version SET version = ?').run(migration.version);
      })();
    } catch (e) {
      throw MIGRATION_ERROR(`Migration ${migration.version} failed: ${(e as Error).message}`);
    }
  }
}
```

## QueryHelpers

```typescript
export function queryOne<T>(sql: string, params: unknown[] = []): T | null {
  const db = getDatabase();
  const stmt = db.prepare(sql);
  const row = stmt.get(...params) as T | undefined;
  return row ?? null;
}

export function queryMany<T>(sql: string, params: unknown[] = []): T[] {
  const db = getDatabase();
  const stmt = db.prepare(sql);
  return stmt.all(...params) as T[];
}

export function execute(sql: string, params: unknown[] = []): void {
  const db = getDatabase();
  const stmt = db.prepare(sql);
  stmt.run(...params);
}

/**
 * Execute function within a transaction with SQLITE_BUSY retry.
 * Uses calculateBackoff() for jittered exponential backoff to prevent thundering herd.
 */
export function transaction<T>(fn: () => T, retries = 3): T {
  const db = getDatabase();

  for (let attempt = 0; attempt <= retries; attempt++) {
    try {
      return db.transaction(fn)();
    } catch (e) {
      const error = e as { code?: string; message: string };

      if (error.code === 'SQLITE_BUSY' && attempt < retries) {
        // Jittered exponential backoff: base * 2^attempt with ±10% jitter
        // Prevents thundering herd when multiple transactions retry simultaneously
        const delay = calculateBackoff(attempt);
        Atomics.wait(new Int32Array(new SharedArrayBuffer(4)), 0, 0, delay);
        continue;
      }

      throw TRANSACTION_ERROR(error.message);
    }
  }

  // Unreachable, but TypeScript needs this
  throw TRANSACTION_ERROR('Transaction retry exhausted');
}
```

```typescript
/**
 * Atomic claim with atomic transition for race protection.
 *
 * CONCURRENCY: Uses two-phase atomic transition pattern:
 * 1. SELECT to verify state and capture updated_at
 * 2. UPDATE with WHERE updated_at = original to detect concurrent modification
 *
 * Returns null if claim fails (not found, wrong state, or race lost).
 */
export function claimAllocation(id: number, runId: number): Allocation | null {
  const db = getDatabase();

  return db.transaction(() => {
    // Phase 1: Read current state
    const allocation = queryOne<Allocation>(
      'SELECT * FROM allocations WHERE id = ? AND status = ? AND run_id IS NULL',
      [id, 'AVAILABLE']
    );

    if (!allocation) {
      // Not found, wrong status, or already claimed
      return null;
    }

    // Phase 2: atomic update with updated_at guard
    const now = Date.now();
    const result = db.prepare(`
      UPDATE allocations
      SET run_id = ?, status = 'CLAIMED', updated_at = ?
      WHERE id = ? AND updated_at = ?
    `).run(runId, now, id, allocation.updated_at);

    if (result.changes === 0) {
      // Race lost - another transaction modified the row
      return null;
    }

    return getAllocation(id);
  })();
}

export function findAvailableAllocation(spec: { spec: string; region?: string }): Allocation | null {
  let sql = `
    SELECT a.* FROM allocations a
    JOIN instances i ON a.instance_id = i.id
    WHERE a.status = 'AVAILABLE' AND a.run_id IS NULL
      AND i.spec = ?
  `;
  const params: unknown[] = [spec.spec];

  if (spec.region) {
    sql += ' AND i.region = ?';
    params.push(spec.region);
  }

  sql += ' LIMIT 1';

  return queryOne<Allocation>(sql, params);
}
```

```typescript
export function sealManifest(id: number): void {
  const db = getDatabase();

  const result = db.prepare(`
    UPDATE manifests SET status = 'SEALED'
    WHERE id = ? AND status = 'DRAFT'
  `).run(id);

  if (result.changes === 0) {
    const manifest = getManifest(id);
    if (!manifest) {
      throw CONSTRAINT_VIOLATION('Manifest not found');
    }
    if (manifest.status === 'SEALED') {
      throw CONSTRAINT_VIOLATION('Manifest already sealed');
    }
  }
}

interface AddResourceOptions {
  cleanupPriority?: number;
  allowRecovery?: boolean;
}

export function addResourceToManifest(
  manifestId: number,
  resourceType: string,
  resourceId: string,
  options: AddResourceOptions = {}
): void {
  const db = getDatabase();
  const now = Date.now();

  const manifest = getManifest(manifestId);
  if (!manifest) {
    throw CONSTRAINT_VIOLATION('Manifest not found');
  }

  // Enforce immutability: SEALED manifests reject additions
  // Exception: A13 recovery path can update provider_id for orphan matching
  if (manifest.status !== 'DRAFT' && !options.allowRecovery) {
    throw CONSTRAINT_VIOLATION('Cannot add resources to sealed manifest');
  }

  db.prepare(`
    INSERT INTO manifest_resources (manifest_id, resource_type, resource_id, cleanup_priority, added_at)
    VALUES (?, ?, ?, ?, ?)
  `).run(
    manifestId,
    resourceType,
    resourceId,
    options.cleanupPriority ?? null,
    now
  );
}

export function getManifestResources(manifestId: number): ManifestResource[] {
  return queryMany<ManifestResource>(
    'SELECT * FROM manifest_resources WHERE manifest_id = ? ORDER BY cleanup_priority DESC',
    [manifestId]
  );
}

export function deleteManifest(id: number): void {
  const db = getDatabase();

  db.transaction(() => {
    // Cascade delete manifest_resources
    db.prepare('DELETE FROM manifest_resources WHERE manifest_id = ?').run(id);
    db.prepare('DELETE FROM manifests WHERE id = ?').run(id);
  })();
}

export function listExpiredManifests(cutoffTime: number): Manifest[] {
  return queryMany<Manifest>(
    'SELECT * FROM manifests WHERE status = ? AND expires_at < ?',
    ['SEALED', cutoffTime]
  );
}
```

```typescript
export function getBlob(id: number): Blob | null {
  return queryOne<Blob>('SELECT * FROM blobs WHERE id = ?', [id]);

/**
 * Create blob with optional CAS deduplication.
 *
 * Dedup is opt-in per bucket:
 * - Dedupable: 'run-files', 'artifacts' - check checksum first
 * - Non-dedupable: 'logs' - always create new
 */
export function createBlob(data: Omit<Blob, 'id' | 'created_at'>): Blob {
  const now = Date.now();
  const db = getDatabase();

  const DEDUPABLE_BUCKETS = ['run-files', 'artifacts'];

  // For dedupable buckets, check if blob exists
  if (DEDUPABLE_BUCKETS.includes(data.bucket) && data.checksum) {
    const existing = findBlobByChecksum(data.bucket, data.checksum);
    if (existing) {
      // Update last_referenced_at and return existing
      updateBlobLastReferenced(existing.id);
      return existing;
    }
  }

  const stmt = db.prepare(`
    INSERT INTO blobs (bucket, checksum, checksum_bytes, s3_key, s3_bucket, payload, size_bytes, created_at, last_referenced_at)
    VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)
  `);

  const result = stmt.run(
    data.bucket,
    data.checksum,
    data.checksum_bytes,
    data.s3_key,
    data.s3_bucket,
    data.payload,
    data.size_bytes,
    now,
    data.last_referenced_at
  );

  return getBlob(result.lastInsertRowid as number)!;
}

export function findBlobByChecksum(bucket: string, checksum: string): Blob | null {
  return queryOne<Blob>(
    'SELECT * FROM blobs WHERE bucket = ? AND checksum = ? LIMIT 1',
```

```typescript
  initChecksum?: string
): Allocation | null {
  let sql = `
    SELECT a.* FROM allocations a
    JOIN instances i ON a.instance_id = i.id
    WHERE a.status = 'AVAILABLE' AND a.run_id IS NULL
      AND i.spec = ?
      AND i.workflow_state LIKE '%:complete'
  `;
  const params: unknown[] = [spec.spec];

  if (spec.region) {
    sql += ' AND i.region = ?';
    params.push(spec.region);
  }

  if (initChecksum) {
    sql += ' AND i.init_checksum = ?';
    params.push(initChecksum);
  }

  sql += ' ORDER BY a.created_at ASC LIMIT 1';

  return queryOne<Allocation>(sql, params);
}

export function getWarmPoolStats(): { available: number; bySpec: Record<string, number> } {
  const db = getDatabase();

  const total = queryOne<{ count: number }>(
    "SELECT COUNT(*) as count FROM allocations WHERE status = 'AVAILABLE'"
  );

  const bySpec = queryMany<{ spec: string; count: number }>(`
    SELECT i.spec || ':' || i.region as spec, COUNT(*) as count
    FROM allocations a
    JOIN instances i ON a.instance_id = i.id
    WHERE a.status = 'AVAILABLE'
    GROUP BY spec
  `);

  return {
    available: total?.count ?? 0,
    bySpec: Object.fromEntries(bySpec.map(r => [r.spec, r.count]))
  };
}
```

```typescript
interface UsageRecord {

export function createUsageRecord(data: Omit<UsageRecord, 'id'>): UsageRecord {
  const db = getDatabase();

  const stmt = db.prepare(`
    INSERT INTO usage_records (instance_id, allocation_id, run_id, provider, spec, region, is_spot, started_at, finished_at, duration_ms, estimated_cost_usd)
    VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
  `);

  const result = stmt.run(
    data.instance_id,
    data.allocation_id,
    data.run_id,
    data.provider,
    data.spec,
    data.region,
    data.is_spot,
    data.started_at,
    data.finished_at,
    data.duration_ms,
    data.estimated_cost_usd
  );

  return queryOne<UsageRecord>('SELECT * FROM usage_records WHERE id = ?', [result.lastInsertRowid])!;
}

export function finishUsageRecord(id: number, finishedAt: number): UsageRecord {
  const db = getDatabase();

  const record = queryOne<UsageRecord>('SELECT * FROM usage_records WHERE id = ?', [id]);
  if (!record) {
    throw CONSTRAINT_VIOLATION('Usage record not found');
  }

  const durationMs = finishedAt - record.started_at;

  db.prepare(`
    UPDATE usage_records
    SET finished_at = ?, duration_ms = ?
    WHERE id = ?
  `).run(finishedAt, durationMs, id);

  return queryOne<UsageRecord>('SELECT * FROM usage_records WHERE id = ?', [id])!;
}

    FROM usage_records
    WHERE started_at >= ? AND started_at < ?
    GROUP BY provider
  `, [monthStart, monthEnd]);
}

export function getActiveUsageRecords(): UsageRecord[] {

  );
}
```

## BackgroundBlobGCOperations

```typescript
/**
 * Find orphaned blobs for garbage collection.
 *
 * Orphaned = no referencing objects + 24h grace period elapsed.
 * Grace period handles in-flight uploads, concurrent runs, and clock skew.
 *
 * Background GC is required because DB CASCADE cannot reach S3 —
 * when objects are deleted, blobs remain and their S3 keys need
 * explicit deletion via provider API.
 */
export function findOrphanedBlobs(cutoff24hAgo: number): Blob[] {
  return queryMany<Blob>(`
    SELECT b.*
    FROM blobs b
    WHERE b.last_referenced_at < ?
      AND NOT EXISTS (
        SELECT 1 FROM objects o
        WHERE o.blob_id = b.id
      )
  `, [cutoff24hAgo]);
}

/**
 * Delete blob records after S3 cleanup.
 * Called by GC after S3 objects have been deleted for blobs with s3_key.
 * Inline blobs (payload in DB) are cleaned up directly.
 */
export function deleteBlobBatch(blobIds: number[]): void {
  if (blobIds.length === 0) return;

  const db = getDatabase();
  const placeholders = blobIds.map(() => '?').join(',');

  db.prepare(`DELETE FROM blobs WHERE id IN (${placeholders})`).run(...blobIds);
}
```
