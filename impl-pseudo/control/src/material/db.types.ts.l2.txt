## RecordTypes

```typescript
import type { ID, TimestampMs, AllocationStatus, ManifestStatus } from '@skyrepl/shared';

// =============================================================================
// RECORD TYPES (Database Rows)
// Conventions:
// - snake_case field names matching SQL columns
// - number for SQLite booleans (0 or 1)
// - string for JSON (parsed on read)
// - All IDs are INTEGER AUTOINCREMENT
// =============================================================================

/** Instance database record */
export interface InstanceRecord {
  id: ID;
  provider: string;
  provider_id: string;
  spec: string;
  region: string | null;
  ip: string | null;
  workflow_state: string;                // Format: <workflow>:<phase>
  workflow_error: string | null;         // JSON-serialized WorkflowError
  current_manifest_id: ID | null;
  spawn_idempotency_key: string | null;  // Format: ${workflow_id}:${node_id}
  is_spot: number;                       // SQLite boolean: 0 or 1
  spot_request_id: string | null;
  init_checksum: string | null;
  created_at: TimestampMs;
  last_heartbeat: TimestampMs;
}

/** Run database record */
export interface RunRecord {
  id: ID;
  command: string;
  workdir: string;
  max_duration_ms: number;
  workflow_state: string;                // Format: <workflow>:<phase>
  workflow_error: string | null;         // JSON-serialized WorkflowError
  current_manifest_id: ID | null;
  exit_code: number | null;
  init_checksum: string | null;
  create_snapshot: number;               // SQLite boolean
  spot_interrupted: number;              // SQLite boolean
  created_at: TimestampMs;
  started_at: TimestampMs | null;
  finished_at: TimestampMs | null;
}

/** Allocation database record */
export interface AllocationRecord {
  id: ID;
  run_id: ID | null;
  instance_id: ID;
  status: AllocationStatus;
  current_manifest_id: ID | null;
  user: string;
  workdir: string;
  debug_hold_until: TimestampMs | null;  // Blocks termination if > NOW
  created_at: TimestampMs;
  updated_at: TimestampMs;
}

/** Workflow database record */
export interface WorkflowRecord {
  id: ID;
  type: string;
  parent_workflow_id: ID | null;
  depth: number;
  status: string;
  current_node: string | null;
  input_json: string;
  output_json: string | null;
  error_json: string | null;
  manifest_id: ID | null;
  trace_id: string | null;
  idempotency_key: string | null;
  timeout_ms: number | null;            // Workflow timeout in milliseconds
  timeout_at: TimestampMs | null;       // Computed: started_at + timeout_ms
  created_at: TimestampMs;
  started_at: TimestampMs | null;
  finished_at: TimestampMs | null;
}

/** Workflow node database record */
export interface WorkflowNodeRecord {
  id: ID;
  workflow_id: ID;
  node_id: string;
  node_type: string;
  status: string;
  input_json: string;
  output_json: string | null;
  error_json: string | null;
  depends_on: string | null;            // JSON array of node_id strings, e.g., '["spawn","prepare"]'
  attempt: number;
  retry_reason: string | null;
  created_at: TimestampMs;
  started_at: TimestampMs | null;
  finished_at: TimestampMs | null;
}

/** Manifest database record */
export interface ManifestRecord {
  id: ID;
  workflow_id: ID;
  status: ManifestStatus;
  default_cleanup_priority: number;
  retention_ms: number | null;
  created_at: TimestampMs;
  released_at: TimestampMs | null;
  expires_at: TimestampMs | null;
}

/** Manifest resource link record */
export interface ManifestResourceRecord {
  manifest_id: ID;
  resource_type: string;
  resource_id: ID;
  cleanup_priority: number | null;
  added_at: TimestampMs;
  owner_type: string;        // 'manifest' | 'workflow' | 'policy'
  owner_id: number | null;   // ID of owner
}

/** Object database record */
export interface ObjectRecord {
  id: ID;
  type: string;
  blob_id: ID;                          // FK to BlobRecord
  provider: string | null;
  provider_object_id: string | null;
  metadata_json: string | null;         // Type-specific metadata (path, content-type, etc.)
  expires_at: TimestampMs | null;
  current_manifest_id: ID | null;
  created_at: TimestampMs;
  accessed_at: TimestampMs | null;
}

/** Blob database record (stored in separate blobs table) */
export interface BlobRecord {
  id: number;
  bucket: string;                    // Tenant bucket identifier
  checksum: string;                  // SHA-256 hash (NOT unique - allows duplicates for append)
  checksum_bytes: number | null;     // Bytes covered by checksum
  size_bytes: number;
  payload: Buffer | null;            // Inline data (<=64KB) or null if in S3
  s3_key: string | null;
  s3_bucket: string | null;
  created_at: TimestampMs;
  last_referenced_at: TimestampMs;
}

/** Run file database record (stored in objects table with type='run_file') */
export interface RunFileRecord {
  id: number;
  type: 'run_file';
  checksum: string;                  // References blob by checksum
  // Tags: run_id:<id>, path:<filepath>
  created_at: TimestampMs;
  updated_at: TimestampMs;
}

/** Provider state cache record */
export interface ProviderStateRecord {
  provider: string;
  key: string;
  value_json: string;
  ttl_at: TimestampMs;
  created_at: TimestampMs;
  updated_at: TimestampMs;
}

/** Usage record for billing (database record) */
export interface UsageRecordDb {
  id: ID;
  instance_id: ID;
  allocation_id: ID | null;
  run_id: ID | null;
  provider: string;
  spec: string;
  region: string | null;
  is_spot: number;                   // SQLite boolean: 0 or 1
  started_at: TimestampMs;
  finished_at: TimestampMs | null;
  duration_ms: number | null;
  estimated_cost_usd: number | null;
}
```

## RecordToMaterializedConverters

```typescript
import type {
  Instance, Run, Allocation, Workflow, Manifest,
  WorkflowNode, Snapshot, Artifact, FeatureInstallation,
  ProviderName, WorkflowExecutionStatus,
} from '@skyrepl/shared';

// =============================================================================
// RECORD -> MATERIALIZED CONVERTERS
// =============================================================================

/** Convert InstanceRecord to Instance (without relationships) */
export function recordToInstance(
  record: InstanceRecord
): Omit<Instance, 'features' | 'tailscaleMachine' | 'externalStatus' | 'externalIp'> {
  return {
    id: record.id,
    provider: record.provider as ProviderName,
    providerId: record.provider_id,
    spec: record.spec,
    region: record.region,
    ip: record.ip,
    workflowState: record.workflow_state,
    workflowError: record.workflow_error,
    isSpot: record.is_spot === 1,
    spotRequestId: record.spot_request_id,
    initChecksum: record.init_checksum,
    spawnIdempotencyKey: record.spawn_idempotency_key,
    createdAt: record.created_at,
    lastHeartbeat: record.last_heartbeat,
    currentManifestId: record.current_manifest_id,
  };
}

/** Convert Instance to InstanceRecord (for insert/update) */
export function instanceToRecord(instance: Instance): InstanceRecord {
  return {
    id: instance.id,
    provider: instance.provider,
    provider_id: instance.providerId,
    spec: instance.spec,
    region: instance.region,
    ip: instance.ip,
    workflow_state: instance.workflowState,
    workflow_error: instance.workflowError,
    current_manifest_id: instance.currentManifestId,
    spawn_idempotency_key: instance.spawnIdempotencyKey,
    is_spot: instance.isSpot ? 1 : 0,
    spot_request_id: instance.spotRequestId,
    init_checksum: instance.initChecksum,
    created_at: instance.createdAt,
    last_heartbeat: instance.lastHeartbeat,
  };
}

/** Convert RunRecord to Run (without relationships) */
export function recordToRun(
  record: RunRecord
): Omit<Run, 'files' | 'logs' | 'artifacts' | 'allocation'> {
  return {
    id: record.id,
    command: record.command,
    workdir: record.workdir,
    maxDurationMs: record.max_duration_ms,
    workflowState: record.workflow_state,
    workflowError: record.workflow_error,
    exitCode: record.exit_code,
    createSnapshot: record.create_snapshot === 1,
    initChecksum: record.init_checksum,
    spotInterrupted: record.spot_interrupted === 1,
    createdAt: record.created_at,
    startedAt: record.started_at,
    finishedAt: record.finished_at,
    currentManifestId: record.current_manifest_id,
  };
}

/** Convert AllocationRecord to Allocation (without relationships) */
export function recordToAllocation(
  record: AllocationRecord
): Omit<Allocation, 'instance' | 'run'> {
  return {
    id: record.id,
    runId: record.run_id,
    instanceId: record.instance_id,
    user: record.user,
    workdir: record.workdir,
    status: record.status,
    debugHoldUntil: record.debug_hold_until,
    createdAt: record.created_at,
    updatedAt: record.updated_at,
    currentManifestId: record.current_manifest_id,
  };
}

/** Convert WorkflowRecord to Workflow (without relationships) */
export function recordToWorkflow(
  record: WorkflowRecord
): Omit<Workflow, 'nodes' | 'manifest'> {
  return {
    id: record.id,
    type: record.type,
    parentWorkflowId: record.parent_workflow_id,
    depth: record.depth,
    status: record.status as WorkflowExecutionStatus,
    currentNode: record.current_node,
    input: JSON.parse(record.input_json),
    output: record.output_json ? JSON.parse(record.output_json) : null,
    error: record.error_json ? JSON.parse(record.error_json) : null,
    manifestId: record.manifest_id,
    traceId: record.trace_id,
    idempotencyKey: record.idempotency_key,
    createdAt: record.created_at,
    startedAt: record.started_at,
    finishedAt: record.finished_at,
  };
}

/** Convert ObjectRecord to Snapshot (for type='snapshot') */
export function recordToSnapshot(record: ObjectRecord): Snapshot {
  const metadata = JSON.parse(record.metadata_json ?? '{}');
  return {
    id: record.id,
    provider: record.provider as ProviderName,
    providerId: record.provider_object_id!,
    spec: metadata.spec,
    initChecksum: metadata.init_checksum,
    name: metadata.name,
    createdAt: record.created_at,
    lastUsedAt: metadata.last_used_at ?? record.created_at,
    useCount: metadata.use_count ?? 0,
  };
}

/** Convert ObjectRecord to Artifact (for type='artifact') */
export function recordToArtifact(record: ObjectRecord, blob?: BlobRecord): Artifact {
  const metadata = JSON.parse(record.metadata_json ?? '{}');
  return {
    id: record.id,
    runId: metadata.run_id,
    path: metadata.path,
    sizeBytes: blob?.size_bytes ?? metadata.size_bytes ?? 0,  // Prefer blob, fall back to metadata
    keep: metadata.keep ?? false,
    createdAt: record.created_at,
  };
}

/** Convert ObjectRecord to FeatureInstallation (for type='feature_installation') */
export function recordToFeatureInstallation(
  record: ObjectRecord
): FeatureInstallation {
  const metadata = JSON.parse(record.metadata_json ?? '{}');
  return {
    id: record.id,
    instanceId: metadata.instance_id,
    featureName: metadata.feature_name,
    version: metadata.version,
    status: metadata.status,
    port: metadata.port ?? null,
    url: metadata.url ?? null,
    installedAt: metadata.installed_at,
  };
}
```

## TypeGuards

```typescript
// =============================================================================
// TYPE GUARDS
// =============================================================================

/** Check if a record looks like an InstanceRecord */
export function isInstanceRecord(record: unknown): record is InstanceRecord {
  return (
    typeof record === 'object' && record !== null &&
    'id' in record && 'provider' in record &&
    'spec' in record && 'workflow_state' in record
  );
}

/** Check if a record looks like a RunRecord */
export function isRunRecord(record: unknown): record is RunRecord {
  return (
    typeof record === 'object' && record !== null &&
    'id' in record && 'command' in record &&
    'workflow_state' in record
  );
}
```
