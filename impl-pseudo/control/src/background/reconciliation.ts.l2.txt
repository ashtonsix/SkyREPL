## MainLoop

```typescript
/**
 * Reconciliation task runs every RECONCILIATION_INTERVAL_MS (60s).
 * Handles: instance state sync, heartbeat timeouts, pending spawns,
 * stuck CLAIMED allocations, and orphan AVAILABLE allocations.
 */
export async function runReconciliationCycle(): Promise<void> {
  const startTime = Date.now();

  try {
    // Run all subtasks
    await syncInstanceState();
    await detectDegradedHeartbeats();  // NEW: Early warning at 2m
    await checkHeartbeatTimeouts();    // UPDATED: 25m with graceful panic detection
    await checkPendingSpawns();
    await recoverStuckClaimedAllocations();
    await cleanupOrphanAvailableAllocations();

    logger.info('reconciliation_cycle_complete', {
      durationMs: Date.now() - startTime
    });
  } catch (error) {
    logger.error('reconciliation_cycle_failed', {
      error: (error as Error).message,
      durationMs: Date.now() - startTime
    });
  }
}

export function startReconciliationLoop(): void {
  setInterval(runReconciliationCycle, TIMING.RECONCILIATION_INTERVAL_MS);
  // Run immediately on startup
  runReconciliationCycle();
}
```

## InstanceStateSync

```typescript
/**
 * Sync instance state from provider APIs.
 * Updates workflow_state on drift.
 */
async function syncInstanceState(): Promise<void> {
  // Group instances by provider
  const instances = queryMany<Instance>(`
    SELECT * FROM instances
    WHERE workflow_state NOT LIKE '%:complete'
      AND workflow_state NOT LIKE '%:error'
  `);

  const byProvider = new Map<string, Instance[]>();
  for (const instance of instances) {
    const list = byProvider.get(instance.provider) || [];
    list.push(instance);
    byProvider.set(instance.provider, list);
  }

  // Query each provider
  for (const [provider, providerInstances] of byProvider) {
    const client = getProviderClient(provider);
    const providerIds = providerInstances.map(i => i.provider_id);

    try {
      const states = await client.getInstanceStates(providerIds);
      const stateMap = new Map(states.map(s => [s.providerId, s]));

      for (const instance of providerInstances) {
        const state = stateMap.get(instance.provider_id);
        await reconcileInstanceState(instance, state);
      }
    } catch (error) {
      logger.warn('provider_sync_failed', {
        provider,
        error: (error as Error).message
      });
      // Continue with other providers
    }
  }
}

async function reconcileInstanceState(
  instance: Instance,
  providerState: ProviderInstanceState | undefined
): Promise<void> {
  if (!providerState || providerState.status === 'terminated') {
    // Instance terminated in provider
    if (!instance.workflow_state.includes('terminate')) {
      logger.warn('instance_terminated_externally', {
        instanceId: instance.id,
        providerId: instance.provider_id
      });
      // Update workflow_state to reflect termination
      transaction(() => {
        queryOne(`
          UPDATE instances SET workflow_state = 'terminate:complete'
          WHERE id = ?
        `, [instance.id]);
      });
    }
  }
}
```

## DegradedHeartbeatDetection

```typescript
/**
 * Detect instances in degraded state (2m since last heartbeat).
 * Provides early warning before the 15m agent panic / 25m force-term.
 *
 * Timeline:
 *   t=0   Last successful heartbeat ack
 *   t=2m  DEGRADED state (this function marks it)
 *   t=15m Agent self-terminates with panic checkpoint
 *   t=25m Control plane force-terminates
 */
async function detectDegradedHeartbeats(): Promise<void> {
  const degradedCutoff = Date.now() - TIMING.HEARTBEAT_DEGRADED_MS;  // 2m
  const staleCutoff = Date.now() - TIMING.STALE_DETECTION_MS;        // 25m

  // Find instances in degraded window (2m-25m since last heartbeat)
  const degradedInstances = queryMany<Instance>(`
    SELECT * FROM instances
    WHERE last_heartbeat < ?
      AND last_heartbeat >= ?
      AND workflow_state NOT LIKE 'degraded:%'
      AND workflow_state NOT LIKE 'suspect:%'
      AND workflow_state NOT LIKE '%:complete'
      AND workflow_state NOT LIKE '%:error'
  `, [degradedCutoff, staleCutoff]);

  for (const instance of degradedInstances) {
    const elapsedMs = Date.now() - instance.last_heartbeat;

    logger.warn('heartbeat_degraded', {
      instanceId: instance.id,
      lastHeartbeat: instance.last_heartbeat,
      degradedDurationMs: elapsedMs
    });

    // Mark as degraded for CLI visibility
    transaction(() => {
      queryOne(`
        UPDATE instances SET workflow_state = 'degraded:control_plane_unreachable'
        WHERE id = ? AND workflow_state NOT LIKE 'degraded:%'
      `, [instance.id]);
    });
  }
}
```

## HeartbeatTimeout

```typescript
/**
 * Find instances with stale heartbeats (25m) and handle termination.
 *
 * IMPORTANT: Before marking as FAILED, check if agent performed graceful
 * panic checkpoint. If checkpoint exists, mark as COMPLETE instead.
 *
 * Detection logic:
 *   1. HeartbeatPanicCompleteMessage received after last_heartbeat_at, OR
 *   2. Checkpoint object created after last_heartbeat_at
 *   If either → COMPLETE (checkpoint preserved)
 *   If neither → FAILED (data potentially lost)
 */
async function checkHeartbeatTimeouts(): Promise<void> {
  const cutoff = Date.now() - TIMING.STALE_DETECTION_MS;  // 25m

  const staleInstances = queryMany<Instance>(`
    SELECT * FROM instances
    WHERE last_heartbeat < ?
      AND workflow_state NOT LIKE '%:complete'
      AND workflow_state NOT LIKE '%:error'
  `, [cutoff]);

  for (const instance of staleInstances) {
    logger.warn('heartbeat_timeout', {
      instanceId: instance.id,
      lastHeartbeat: instance.last_heartbeat,
      staleDurationMs: Date.now() - instance.last_heartbeat
    });

    // Get active allocations for this instance
    const allocations = queryMany<Allocation>(`
      SELECT * FROM allocations
      WHERE instance_id = ? AND status = 'ACTIVE'
    `, [instance.id]);

    for (const allocation of allocations) {
      // Check for graceful panic checkpoint
      const gracefulPanic = await checkGracefulPanic(allocation.id, instance.last_heartbeat);

      if (gracefulPanic.hasCheckpoint) {
        // Agent checkpointed before dying - graceful termination
        logger.info('allocation_complete_graceful_panic', {
          allocationId: allocation.id,
          instanceId: instance.id,
          checkpointAt: gracefulPanic.checkpointAt
        });

        await completeAllocation(allocation.id, 'ACTIVE', {
          reason: 'heartbeat_panic_graceful',
          checkpoint_at: gracefulPanic.checkpointAt
        });
      } else {
        // No checkpoint - crash, data potentially lost
        logger.error('allocation_failed_heartbeat_timeout', {
          allocationId: allocation.id,
          instanceId: instance.id
        });

        await failAllocation(allocation.id, 'ACTIVE');
      }
    }

    // Mark instance as suspect
    transaction(() => {
      queryOne(`
        UPDATE instances SET workflow_state = 'suspect:heartbeat_timeout'
        WHERE id = ?
      `, [instance.id]);
    });
  }
}

interface GracefulPanicResult {
  hasCheckpoint: boolean;
  checkpointAt?: number;
}

/**
 * Check if agent performed graceful panic checkpoint.
 * Looks for:
 *   1. HeartbeatPanicCompleteMessage after last_heartbeat
 *   2. Checkpoint object created after last_heartbeat
 */
async function checkGracefulPanic(
  allocationId: number,
  lastHeartbeat: number
): Promise<GracefulPanicResult> {
  // Check for HeartbeatPanicCompleteMessage stored in objects table
  const panicComplete = queryMany<{ created_at: number }>(`
    SELECT o.created_at
    FROM objects o
    WHERE o.type = 'message'
      AND JSON_EXTRACT(o.metadata_json, '$.allocation_id') = ?
      AND JSON_EXTRACT(o.metadata_json, '$.message_type') = 'heartbeat_panic_complete'
      AND o.created_at > ?
    ORDER BY o.created_at DESC
    LIMIT 1
  `, [allocationId, lastHeartbeat]);

  if (panicComplete.length > 0) {
    return { hasCheckpoint: true, checkpointAt: panicComplete[0].created_at };
  }

  // Check for checkpoint artifacts created after last heartbeat
  const checkpoints = queryMany<{ created_at: number }>(`
    SELECT o.created_at
    FROM objects o
    WHERE o.type = 'artifact'
      AND JSON_EXTRACT(o.metadata_json, '$.allocation_id') = ?
      AND o.created_at > ?
      AND (JSON_EXTRACT(o.metadata_json, '$.artifact_type') = 'checkpoint'
           OR JSON_EXTRACT(o.metadata_json, '$.path') LIKE '%checkpoint%')
    ORDER BY o.created_at DESC
    LIMIT 1
  `, [allocationId, lastHeartbeat]);

  if (checkpoints.length > 0) {
    return { hasCheckpoint: true, checkpointAt: checkpoints[0].created_at };
  }

  return { hasCheckpoint: false };
}
```

## PendingSpawnHandling

```typescript
/**
 * Check spawn:pending instances for completion or failure.
 */
async function checkPendingSpawns(): Promise<void> {
  const pendingInstances = queryMany<Instance>(`
    SELECT * FROM instances
    WHERE workflow_state = 'spawn:pending'
  `);

  for (const instance of pendingInstances) {
    const client = getProviderClient(instance.provider);

    try {
      const state = await client.getInstanceState(instance.provider_id);

      if (state.status === 'running') {
        // Spawn completed
        transaction(() => {
          queryOne(`
            UPDATE instances SET workflow_state = 'spawn:complete'
            WHERE id = ?
          `, [instance.id]);
        });
        logger.info('spawn_completed', { instanceId: instance.id });
      } else if (state.status === 'terminated' || state.status === 'unknown') {
        // Spawn failed
        transaction(() => {
          queryOne(`
            UPDATE instances SET workflow_state = 'spawn:error'
            WHERE id = ?
          `, [instance.id]);
        });
        logger.error('spawn_failed', {
          instanceId: instance.id,
          providerStatus: state.status
        });
      }
      // If still pending, do nothing - check again next cycle
    } catch (error) {
      logger.warn('spawn_check_failed', {
        instanceId: instance.id,
        error: (error as Error).message
      });
    }
  }
}
```

## StuckClaimedAllocationRecovery

```typescript
/**
 * Recover allocations stuck in CLAIMED state.
 *
 * CLAIMED timeout: 10 minutes (sync should complete well within this)
 * Cause: Control plane crash after agent sync_complete but before DB update
 *
 * CRITICAL: Uses failAllocation(id, 'CLAIMED') with explicit status guard
 * to prevent accidentally failing an allocation that transitioned to ACTIVE.
 */
async function recoverStuckClaimedAllocations(): Promise<void> {
  const cutoff = Date.now() - TIMING.CLAIMED_TIMEOUT_MS;

  const stuckAllocations = queryMany<Allocation>(`
    SELECT * FROM allocations
    WHERE status = 'CLAIMED' AND updated_at < ?
  `, [cutoff]);

  for (const allocation of stuckAllocations) {
    const elapsed = Date.now() - allocation.updated_at;

    // CRITICAL: Use atomic transition with status guard
    // This prevents race condition where allocation transitions CLAIMED -> ACTIVE
    // between our query and update
    const result = await failAllocation(allocation.id, 'CLAIMED');

    if (result.success) {
      logger.info('recovered_stuck_claimed', {
        allocationId: allocation.id,
        elapsedMs: elapsed
      });

      // Mark instance as suspect
      transaction(() => {
        queryOne(`
          UPDATE instances SET workflow_state = 'suspect:claimed_timeout'
          WHERE id = (SELECT instance_id FROM allocations WHERE id = ?)
        `, [allocation.id]);
      });
    } else {
      // Transition failed - allocation state changed
      switch (result.reason) {
        case 'NOT_FOUND':
          logger.debug('stuck_claimed_not_found', { allocationId: allocation.id });
          break;

        case 'WRONG_STATE':
          // Allocation progressed normally (e.g., to ACTIVE)
          logger.debug('stuck_claimed_state_changed', {
            allocationId: allocation.id,
            currentStatus: result.current?.status
          });
          break;

        case 'RACE_LOST':
          // Another process handled it
          logger.debug('stuck_claimed_race', {
            allocationId: allocation.id,
            currentStatus: result.current?.status
          });
          break;
      }
    }
  }
}
```

## OrphanAvailableAllocationCleanup

```typescript
/**
 * Clean up AVAILABLE allocations referencing dead instances.
 *
 * Purpose: Prevents warm pool pollution where AVAILABLE allocations
 * reference terminated instances. Without this, claim attempts would
 * fail repeatedly, creating an infinite loop.
 *
 * CRITICAL: Uses failAllocation(id, 'AVAILABLE') with explicit status guard
 * to prevent accidentally failing an allocation that was claimed between
 * our query and update.
 */
async function cleanupOrphanAvailableAllocations(): Promise<void> {
  const availableAllocations = queryMany<Allocation & { instance_id: number }>(`
    SELECT a.*, a.instance_id
    FROM allocations a
    WHERE a.status = 'AVAILABLE'
  `);

  // Group by provider for efficient API calls
  const instanceIds = [...new Set(availableAllocations.map(a => a.instance_id))];
  const instances = queryMany<Instance>(`
    SELECT * FROM instances WHERE id IN (${instanceIds.map(() => '?').join(',')})
  `, instanceIds);
  const instanceMap = new Map(instances.map(i => [i.id, i]));

  // Group by provider
  const byProvider = new Map<string, Instance[]>();
  for (const instance of instances) {
    const list = byProvider.get(instance.provider) || [];
    list.push(instance);
    byProvider.set(instance.provider, list);
  }

  // Get provider states
  const providerStates = new Map<string, ProviderInstanceState>();
  for (const [provider, providerInstances] of byProvider) {
    const client = getProviderClient(provider);
    try {
      const states = await client.getInstanceStates(providerInstances.map(i => i.provider_id));
      for (const state of states) {
        providerStates.set(state.providerId, state);
      }
    } catch (error) {
      logger.warn('orphan_check_provider_failed', {
        provider,
        error: (error as Error).message
      });
    }
  }

  // Check each allocation
  for (const allocation of availableAllocations) {
    const instance = instanceMap.get(allocation.instance_id);
    if (!instance) {
      // Instance record deleted - fail allocation
      await failOrphanAllocation(allocation, 'instance record not found');
      continue;
    }

    const providerState = providerStates.get(instance.provider_id);

    if (!providerState) {
      // Instance not found in provider
      await failOrphanAllocation(allocation, `instance ${instance.id} not found in provider`);
    } else if (providerState.status === 'terminated') {
      // Instance terminated in provider
      await failOrphanAllocation(allocation, `instance ${instance.id} terminated`);
    }
  }
}

async function failOrphanAllocation(
  allocation: Allocation,
  reason: string
): Promise<void> {
  // CRITICAL: Use atomic transition with status guard
  // This prevents race condition where allocation is claimed between
  // our query and update
  const result = await failAllocation(allocation.id, 'AVAILABLE');

  if (result.success) {
    logger.info('marked_orphan_available_failed', {
      allocationId: allocation.id,
      reason
    });
  } else if (result.reason === 'WRONG_STATE') {
    // Allocation was claimed - this is fine
    logger.debug('orphan_allocation_claimed', {
      allocationId: allocation.id,
      currentStatus: result.current?.status
    });
  }
  // RACE_LOST or NOT_FOUND are also fine - another process handled it
}
```

## ErrorHandling

```typescript
/**
 * Error handling policy for reconciliation.
 *
 * - Transient API failures: Log and retry next cycle
 * - Persistent failures: Alert after N consecutive failures
 */
const consecutiveFailures = new Map<string, number>();
const ALERT_THRESHOLD = 5;

function recordFailure(task: string): void {
  const count = (consecutiveFailures.get(task) || 0) + 1;
  consecutiveFailures.set(task, count);

  if (count >= ALERT_THRESHOLD) {
    logger.error('reconciliation_persistent_failure', {
      task,
      consecutiveFailures: count
    });
    // TODO: Emit alert metric
  }
}

function recordSuccess(task: string): void {
  consecutiveFailures.delete(task);
}
```
