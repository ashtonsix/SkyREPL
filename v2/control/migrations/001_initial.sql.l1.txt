## objects
- id INTEGER PRIMARY KEY AUTOINCREMENT
- type TEXT NOT NULL
- provider TEXT NOT NULL
- provider_object_id TEXT
- size_bytes INTEGER NOT NULL
- reservation INTEGER
  # Optional. Invariant: reservation >= size_bytes when set.
  # If object is expected to grow beyond 4KB, set reservation to expected final size.
  # This causes immediate S3 allocation (skip inline storage).
- content_type TEXT
- payload BLOB
- s3_key TEXT
- s3_bucket TEXT
- checksum TEXT
  # Content hash (sha256). NULL if not yet computed or not applicable.
  # For content-addressable blobs, used for deduplication via idx_objects_checksum.
- checksum_bytes INTEGER
  # Number of bytes covered by checksum. When checksum_bytes == size_bytes, checksum is current.
  # When checksum_bytes < size_bytes (after append operations), checksum is stale.
  # NULL when checksum is NULL.
- expires_at INTEGER
- current_manifest_id INTEGER
- created_at INTEGER NOT NULL
- accessed_at INTEGER
Indexes: idx_objects_manifest, idx_objects_checksum, idx_objects_expires, idx_objects_type, UNIQUE(provider, provider_object_id)

## object_tags
- object_id INTEGER NOT NULL
- tag TEXT NOT NULL
PRIMARY KEY: (object_id, tag)
FOREIGN KEY: object_id REFERENCES objects(id) ON DELETE CASCADE
Indexes: idx_object_tags_tag

# NOTE: There is no separate 'blobs' table. Blobs are stored as objects with
# type='blob'. The checksum column on objects enables deduplication (query
# by checksum before insert). For blob garbage collection, query objects
# WHERE type='blob' that have no referencing run_file objects.
