## Control -> Agent Messages (SSE)

```typescript
import type { ID, TimestampMs, TailscaleStatus } from '../types';

// =============================================================================
// CONTROL -> AGENT (sent via SSE stream)
// =============================================================================

/** Start a run on the instance */
export interface StartRunMessage {
  type: 'start_run';
  command_id: number;         // Required: for acknowledgment protocol
  run_id: ID;
  allocation_id: ID;
  command: string;
  workdir: string;
  files: AgentFileEntry[];
  artifacts: string[];        // Glob patterns to collect (e.g., ["**/*.pt"])
  run_init?: boolean;         // Run init script before command
  init_script?: string;       // Init script content (if run_init=true)
  create_snapshot?: boolean;  // Create snapshot after run completes
}

/** File entry for agent download */
export interface AgentFileEntry {
  path: string;               // Relative path in workdir
  checksum: string;           // SHA256 checksum for verification
  url: string;                // Presigned S3 URL or control plane URL
  failover_url?: string;      // Secondary presigned URL for resilience
}

/** Cancel a running run */
export interface CancelRunMessage {
  type: 'cancel_run';
  command_id: number;         // Required: for acknowledgment protocol
  run_id: ID;
}

/** Prepare instance for snapshot */
export interface PrepareSnapshotMessage {
  type: 'prepare_snapshot';
  command_id: number;         // Required: for acknowledgment protocol
  instance_id: ID;
}

/** Capture artifacts from a run */
export interface CaptureArtifactsMessage {
  type: 'capture_artifacts';
  command_id?: number;        // Optional (fire-and-forget acceptable)
  run_id: ID;
}

/** Trigger Tailscale installation */
export interface TriggerTailscaleMessage {
  type: 'trigger_tailscale';
  command_id?: number;        // Optional (fire-and-forget acceptable)
}

/** Acknowledge agent heartbeat */
export interface HeartbeatAckMessage {
  type: 'heartbeat_ack';
  control_plane_id: string;   // Unique control plane identifier
}

/** Acknowledge snapshot preparation */
export interface PrepareSnapshotAckMessage {
  type: 'prepare_snapshot_ack';
  instance_id: ID;
  success: boolean;
  error?: string;
}

/** Union of all control -> agent message types */
export type ControlToAgentMessage =
  | StartRunMessage
  | CancelRunMessage
  | PrepareSnapshotMessage
  | CaptureArtifactsMessage
  | TriggerTailscaleMessage
  | HeartbeatAckMessage
  | PrepareSnapshotAckMessage;
```

## Agent -> Control Messages (HTTP POST)

```typescript
// =============================================================================
// AGENT -> CONTROL (sent via HTTP POST)
// =============================================================================

/** Log output from a run */
export interface LogMessage {
  type: 'log';
  run_id: ID;
  allocation_id: ID;
  stream: 'stdout' | 'stderr';
  data: string;
  phase?: 'setup' | 'execution' | 'teardown';
  timestamp: TimestampMs;
  batch_sequence_id: number;  // Monotonic counter per agent session for ordering
}

/** Run completed */
export interface RunCompleteMessage {
  type: 'run_complete';
  run_id: ID;
  allocation_id: ID;
  exit_code: number;
  workflow_state: Record<string, unknown>;
}

/** Agent heartbeat */
export interface HeartbeatMessage {
  type: 'heartbeat';
  tailscale_ip: string;
  active_ssh_sessions: number;
  tailscale_status: TailscaleStatus;
  workflow_state: Record<string, unknown>;
  pending_command_acks: number[];  // Command IDs to acknowledge
  dropped_logs_count?: number;    // Logs dropped due to buffer overflow
}

/** Spot interrupt detected - Phase 1 (immediate notification) */
export interface SpotInterruptStartMessage {
  type: 'spot_interrupt_start';
  run_id: ID;
  time_remaining_ms: number;  // Time until forced termination (from IMDS)
}

/** Spot interrupt - Phase 2 (checkpoint complete) */
export interface SpotInterruptCompleteMessage {
  type: 'spot_interrupt_complete';
  run_id: ID;
  checkpoint_exit_code?: number;  // 0=success, non-zero=failure, undefined=not run
  artifacts_uploaded: number;
}

/** Heartbeat panic detected - Phase 1 (immediate notification) */
export interface HeartbeatPanicStartMessage {
  type: 'heartbeat_panic_start';
  run_id: ID;
  time_since_ack_ms: number;  // Time since last successful heartbeat ack (~15m)
}

/** Heartbeat panic - Phase 2 (checkpoint complete) */
export interface HeartbeatPanicCompleteMessage {
  type: 'heartbeat_panic_complete';
  run_id: ID;
  checkpoint_exit_code?: number;
  artifacts_uploaded: number;
}

/** Agent panic (critical failure) */
export interface PanicRequest {
  type: 'panic';
  instance_id: ID;
  reason: string;
  last_state_json: string;
  diagnostics_json: string;
  error_logs_json: string;
  timestamp: TimestampMs;
}

/** Union of all agent -> control message types */
export type AgentToControlMessage =
  | LogMessage
  | RunCompleteMessage
  | HeartbeatMessage
  | SpotInterruptStartMessage
  | SpotInterruptCompleteMessage
  | HeartbeatPanicStartMessage
  | HeartbeatPanicCompleteMessage
  | PanicRequest;
```

## Command Acknowledgment Protocol

```typescript
// =============================================================================
// COMMAND ACKNOWLEDGMENT PROTOCOL
// =============================================================================

/**
 * Commands requiring acknowledgment (command_id required):
 * - StartRunMessage
 * - CancelRunMessage
 * - PrepareSnapshotMessage
 *
 * Commands with optional ack (fire-and-forget acceptable):
 * - CaptureArtifactsMessage
 * - TriggerTailscaleMessage
 *
 * Flow:
 * 1. Control sends command via SSE with command_id (autoincrement integer)
 * 2. Agent adds command_id to pending_command_acks set
 * 3. Agent processes command, adds to processed_command_ids (last 100)
 * 4. Agent includes pending_command_acks in next heartbeat
 * 5. Control marks commands as acknowledged
 * 6. Agent clears pending_command_acks after successful heartbeat
 *
 * Timeout/Retry (control plane):
 * - Timeout: 30s from command send
 * - Max retries: 3 attempts
 * - After max retries: mark command as failed, log error, continue
 *
 * Deduplication (agent):
 * - Tracks last 100 command_ids in circular buffer
 * - Duplicate command_id: skip execution, add to pending_command_acks
 */
```
