# heartbeat.py - Heartbeat Thread (L2 Pseudocode)

# Note: Timing constants reference ../../shared/src/config/timing.ts.l2.txt
# - HEARTBEAT_INTERVAL_MS (10s)
# - HEARTBEAT_DEGRADED_MS (2m) - early warning threshold
# - HEARTBEAT_ACK_TIMEOUT_MS (15m) - agent self-termination trigger
# - PANIC_CHECKPOINT_BUDGET_MS (5m) - checkpoint budget during panic

```python
import threading
import time
from typing import Optional
from checkpoint import execute_panic_checkpoint
from artifacts import get_artifact_urls

# State
last_ack_time: float = time.time()
degraded_since: Optional[float] = None
shutting_down: bool = False
expected_control_plane_id: Optional[str] = None  # Set from first heartbeat ack


def heartbeat_thread():
    global last_ack_time, degraded_since, shutting_down, expected_control_plane_id

    # CRITICAL: Immediate heartbeat before first sleep
    if send_heartbeat():
        last_ack_time = time.time()
        log("INFO", "Initial heartbeat acknowledged")
    else:
        log("WARN", "Initial heartbeat failed")

    while not shutting_down:
        time.sleep(HEARTBEAT_INTERVAL_MS / 1000.0)  # 10s
        if shutting_down:
            break

        time_since_ack_ms = (time.time() - last_ack_time) * 1000

        # Check degraded threshold (2m)
        if time_since_ack_ms >= HEARTBEAT_DEGRADED_MS and degraded_since is None:
            degraded_since = time.time()
            log("WARN", f"Control plane unreachable for {HEARTBEAT_DEGRADED_MS/1000}s, entering DEGRADED state")

        # Check panic threshold (15m)
        if time_since_ack_ms >= HEARTBEAT_ACK_TIMEOUT_MS:
            log("ERROR", f"Control plane unreachable for {HEARTBEAT_ACK_TIMEOUT_MS/1000}s, initiating panic checkpoint")
            initiate_panic_checkpoint(time_since_ack_ms)
            return  # Thread exits, shutdown initiated

        # Build workflow_state based on degraded status
        workflow_state = get_current_workflow_state()
        if degraded_since is not None:
            workflow_state = 'degraded:control_plane_unreachable'

        # Send heartbeat
        if send_heartbeat(workflow_state):
            # Ack received - reset state
            last_ack_time = time.time()
            if degraded_since is not None:
                log("INFO", f"Control plane recovered after {time.time() - degraded_since:.1f}s degraded")
                degraded_since = None


def send_heartbeat(workflow_state: str = None) -> bool:
    """POST /v1/agent/heartbeat. Returns True if ack received."""
    global expected_control_plane_id

    payload = {
        'tailscale_ip': get_tailscale_ip(),
        'active_allocations': get_active_allocations_with_ssh(),  # [{allocation_id, has_ssh_sessions}]
        'tailscale_status': get_tailscale_status(),
        'workflow_state': workflow_state or get_current_workflow_state(),
        'pending_command_acks': get_pending_command_acks(),
        'dropped_logs_count': get_dropped_logs_count(),
    }

    try:
        response = http_post('/v1/agent/heartbeat', payload, timeout=10)
        if response.status == 200:
            response_data = response.json()

            # Validate control_plane_id
            received_id = response_data.get('control_plane_id')
            if received_id:
                if expected_control_plane_id is None:
                    # First successful heartbeat - store ID
                    expected_control_plane_id = received_id
                    log("INFO", f"Control plane ID established: {received_id}")
                elif received_id != expected_control_plane_id:
                    # Mismatch detected - log warning but continue
                    log("WARN", f"Control plane ID mismatch: expected={expected_control_plane_id}, received={received_id}")

            process_heartbeat_response(response_data)
            return True
        log("WARN", f"Heartbeat returned {response.status}")
        return False
    except Exception as e:
        log("WARN", f"Heartbeat failed: {e}")
        return False


def get_active_allocations_with_ssh() -> list:
    """Build per-allocation SSH state for heartbeat payload.
    Returns list of {allocation_id: int, has_ssh_sessions: bool}."""
    result = []
    for alloc_id, user in get_tracked_allocations():
        # `who | grep -c "<user>"` per allocation to detect active SSH sessions
        session_count = count_ssh_sessions_for_user(user)
        result.append({
            'allocation_id': alloc_id,
            'has_ssh_sessions': session_count > 0,
        })
    return result


def initiate_panic_checkpoint(time_since_ack_ms: int):
    """Execute panic checkpoint and shutdown. Reuses spot interrupt checkpoint path."""
    global shutting_down
    shutting_down = True

    run_id = get_current_run_id()

    # Phase 1: Signal panic start
    send_panic_start_message(run_id, time_since_ack_ms)

    # Get artifact URLs for upload (same pattern as spot interrupt)
    artifact_urls = get_artifact_urls()

    # Execute checkpoint (5m budget)
    checkpoint_result = execute_panic_checkpoint(
        run_id=run_id,
        artifact_urls=artifact_urls,
    )

    # Phase 2: Signal panic complete
    send_panic_complete_message(
        run_id=run_id,
        checkpoint_exit_code=checkpoint_result.exit_code,
        artifacts_uploaded=checkpoint_result.artifacts_uploaded,
    )

    # Initiate shutdown (same path as spot/SIGTERM)
    initiate_shutdown()


def send_panic_start_message(run_id: int, time_since_ack_ms: int):
    """POST /v1/agent/heartbeat-panic-start"""
    payload = {
        'type': 'heartbeat_panic_start',
        'run_id': run_id,
        'time_since_ack_ms': time_since_ack_ms,
    }
    try:
        http_post('/v1/agent/heartbeat-panic-start', payload, timeout=5)
    except Exception as e:
        # Best-effort - control plane may be unreachable
        log("WARN", f"Failed to send panic start: {e}")


def send_panic_complete_message(run_id: int, checkpoint_exit_code: Optional[int], artifacts_uploaded: int):
    """POST /v1/agent/heartbeat-panic-complete"""
    payload = {
        'type': 'heartbeat_panic_complete',
        'run_id': run_id,
        'checkpoint_exit_code': checkpoint_exit_code,
        'artifacts_uploaded': artifacts_uploaded,
    }
    try:
        http_post('/v1/agent/heartbeat-panic-complete', payload, timeout=5)
    except Exception as e:
        # Best-effort - control plane may be unreachable
        log("WARN", f"Failed to send panic complete: {e}")
```

## TimingBudget

- Checkpoint script: PANIC_CHECKPOINT_BUDGET_MS (5m)
- Artifact upload: 4m
- Messaging overhead: 1m buffer
- Control plane force-term at 25m if agent fails

## SharedLogicWithSpotInterrupt

The `execute_panic_checkpoint()` function is shared between:
- Spot interruption (2m warning, 90s checkpoint budget)
- Heartbeat panic (10m window, 5m checkpoint budget)

Both paths:
1. Set SKYREPL_CHECKPOINT_REASON env var
2. Execute user checkpoint script with timeout
3. Upload artifacts in priority order
4. Send completion message

See checkpoint.py.l2.txt for shared implementation.

## ControlPlaneRecovery

If control plane recovers during degraded window (2m-15m):
- Agent receives heartbeat ack
- Clears degraded_since, resumes normal state
- workflow_state returns to normal

This prevents unnecessary panic checkpoints when control plane has transient issues.
