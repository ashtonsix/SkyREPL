"""
Artifact Collection and Upload

Collects files matching glob patterns from workdir, uploads to presigned URLs.
Presigned URL abstraction (spec 11.11):
  - >64KB: S3 directly (control plane never sees payload)
  - <=64KB: Control plane endpoint (lower latency)
Agent code unchanged - same PUT regardless of backing store.

Checkpoint-related artifact upload (with prioritization and failover)
is handled in checkpoint.py.l2.txt. This module handles normal
post-run artifact collection.
"""

```python
import fnmatch
import hashlib
import os
import time
import urllib.request
from typing import List, Optional


# =============================================================================
# Glob Matching
# =============================================================================

def match_artifact_patterns(workdir: str, patterns: List[str]) -> List[str]:
    """
    Match artifact glob patterns against workdir contents.

    Walks workdir recursively, matches each relative path against patterns.
    Uses fnmatch for glob matching (supports *, ?, []).

    Args:
        workdir: Base directory to search
        patterns: Glob patterns from StartRunMessage.artifacts
                  e.g., ["**/*.pt", "logs/*.txt", "output.json"]

    Returns:
        List of relative paths matching any pattern (deduplicated)
    """
    matched = set()

    for root, dirs, files in os.walk(workdir):
        for filename in files:
            full_path = os.path.join(root, filename)
            rel_path = os.path.relpath(full_path, workdir)

            for pattern in patterns:
                if fnmatch.fnmatch(rel_path, pattern):
                    matched.add(rel_path)
                    break  # No need to check other patterns

    result = sorted(matched)
    log("INFO", f"Matched {len(result)} artifacts from {len(patterns)} patterns")
    return result


# =============================================================================
# Artifact Upload
# =============================================================================

def upload_artifact(workdir: str, rel_path: str, presigned_url: str) -> Optional[dict]:
    """
    Upload single artifact to presigned URL.

    Presigned URL abstraction:
    - Agent just does urllib.request.Request(url, data=file_data, method="PUT")
    - URL may point to S3 (>64KB) or control plane endpoint (<=64KB)

    Args:
        workdir: Base directory
        rel_path: Relative path of artifact
        presigned_url: Presigned S3 or control plane URL

    Returns:
        ArtifactMessage dict on success, None on failure
    """
    full_path = os.path.join(workdir, rel_path)

    try:
        # Read file and compute checksum
        with open(full_path, 'rb') as f:
            data = f.read()

        checksum = hashlib.sha256(data).hexdigest()
        size_bytes = len(data)

        # Upload via presigned URL
        req = urllib.request.Request(
            presigned_url,
            data=data,
            method="PUT",
        )
        with urllib.request.urlopen(req, timeout=60) as resp:
            if resp.status not in (200, 204):
                log("WARN", f"Artifact upload returned {resp.status}: {rel_path}")
                return None

        log("INFO", f"Uploaded artifact: {rel_path} ({size_bytes} bytes)")

        return {
            'type': 'artifact',
            'path': rel_path,
            'checksum': checksum,
            'size_bytes': size_bytes,
        }

    except Exception as e:
        log("ERROR", f"Artifact upload failed for {rel_path}: {e}")
        return None


# =============================================================================
# Collection and Upload Pipeline
# =============================================================================

def collect_and_upload_artifacts(run_id: str, workdir: str, patterns: List[str]) -> int:
    """
    Full artifact pipeline: match patterns, request URLs, upload, confirm.

    Sequence:
    1. Match artifact patterns against workdir
    2. Request presigned upload URLs from control plane
    3. Upload each artifact
    4. Send ArtifactMessage confirmation for each

    Args:
        run_id: Current run ID
        workdir: Run working directory
        patterns: Glob patterns from StartRunMessage.artifacts

    Returns:
        Count of successfully uploaded artifacts
    """
    if not patterns:
        return 0

    # Step 1: Match patterns
    matched_paths = match_artifact_patterns(workdir, patterns)
    if not matched_paths:
        log("INFO", f"No artifacts matched for run {run_id}")
        return 0

    # Step 2: Request presigned upload URLs
    upload_urls = request_artifact_urls(run_id, matched_paths)
    if not upload_urls:
        log("ERROR", f"Failed to get artifact upload URLs for run {run_id}")
        return 0

    # Step 3: Upload each artifact
    uploaded = 0
    for rel_path in matched_paths:
        url_info = upload_urls.get(rel_path)
        if not url_info:
            log("WARN", f"No upload URL for artifact: {rel_path}")
            continue

        result = upload_artifact(workdir, rel_path, url_info['url'])
        if result:
            # Step 4: Send confirmation
            result['run_id'] = run_id
            result['s3_key'] = url_info.get('s3_key', '')
            send_artifact_message(result)
            uploaded += 1

    log("INFO", f"Uploaded {uploaded}/{len(matched_paths)} artifacts for run {run_id}")
    return uploaded


def request_artifact_urls(run_id: str, paths: List[str]) -> Optional[dict]:
    """
    Request presigned upload URLs from control plane.

    POST /v1/agent/artifact-urls
    Body: { run_id, paths: [...] }
    Response: { path: { url, s3_key, failover_url? }, ... }
    """
    payload = {
        'run_id': run_id,
        'paths': paths,
    }
    try:
        response = http_post('/v1/agent/artifact-urls', payload, timeout=30)
        if response.status == 200:
            return response.json()
        log("ERROR", f"Artifact URL request failed: {response.status}")
        return None
    except Exception as e:
        log("ERROR", f"Artifact URL request error: {e}")
        return None


def send_artifact_message(artifact_info: dict) -> None:
    """
    POST /v1/agent/artifacts - Confirm artifact upload.

    Payload: ArtifactMessage { type, run_id, path, s3_key, size_bytes }
    """
    try:
        http_post('/v1/agent/artifacts', artifact_info, timeout=10)
    except Exception as e:
        log("WARN", f"Failed to send artifact confirmation: {e}")


# =============================================================================
# Logging Utility
# =============================================================================

def log(level: str, message: str) -> None:
    """Internal logging."""
    timestamp = time.strftime('%Y-%m-%d %H:%M:%S')
    print(f"[{timestamp}] [{level}] [artifacts] {message}")
```
