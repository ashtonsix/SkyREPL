## objects
- id INTEGER PRIMARY KEY AUTOINCREMENT
- type TEXT NOT NULL
  # Object types: 'artifact', 'snapshot', 'log', 'run_file' (NOT 'blob' - blobs are separate table)
- blob_id INTEGER NOT NULL
  # Foreign key to blobs table. All objects use blobs for payload storage.
- provider TEXT NOT NULL
- provider_object_id TEXT
  # Provider-specific identifier (e.g., snapshot ID, artifact ID)
- metadata_json TEXT
  # Type-specific metadata (filename, content-type, etc.)
- expires_at INTEGER
- current_manifest_id INTEGER
- created_at INTEGER NOT NULL
- accessed_at INTEGER
Indexes: idx_objects_manifest, idx_objects_blob, idx_objects_expires, idx_objects_type, UNIQUE(provider, provider_object_id)
FOREIGN KEY: blob_id REFERENCES blobs(id) ON DELETE RESTRICT
# ON DELETE RESTRICT: prevents blob deletion while objects reference it (GC handles this)

## blobs
- id INTEGER PRIMARY KEY AUTOINCREMENT
- bucket TEXT NOT NULL
  # Logical grouping for blob collections (e.g., 'run-files', 'artifacts', 'logs')
- checksum TEXT NOT NULL
  # Content hash (sha256). NOT unique - logs use append, creating multiple blobs with same checksum prefix.
- checksum_bytes INTEGER
  # Bytes covered by checksum. NULL when checksum not computed.
  # When checksum_bytes == size_bytes: checksum is current
  # When checksum_bytes < size_bytes: checksum is stale (after append, needs recomputation)
- s3_key TEXT
  # S3 object key. NULL if inline (SBO).
- s3_bucket TEXT
  # S3 bucket name. NULL if inline (SBO).
- payload BLOB
  # Inline storage for small blobs (≤64KB). NULL if stored in S3.
- size_bytes INTEGER NOT NULL
- created_at INTEGER NOT NULL
- last_referenced_at INTEGER NOT NULL
  # Updated when objects reference this blob. Used by GC to identify stale blobs.
Indexes: idx_blobs_dedup (bucket, checksum), idx_blobs_last_referenced
# CAS dedup is OPTIONAL: opt-in per object type via idx_blobs_dedup, not schema constraint
# Background GC finds blobs with no referencing objects (DB cascade can't reach S3)
