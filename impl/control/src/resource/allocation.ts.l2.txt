## ValidTransitionsMap

```typescript
const VALID_TRANSITIONS: Record<AllocationStatus, AllocationStatus[]> = {
  'AVAILABLE': ['CLAIMED'],
  'CLAIMED': ['ACTIVE', 'FAILED'],  // ACTIVE triggered by sync_complete
  'ACTIVE': ['COMPLETE', 'FAILED'],
  'COMPLETE': [],  // Terminal state (eventually archived)
  'FAILED': [],    // Terminal state (instance terminated)
};

const TRANSITION_TRIGGERS = {
  'AVAILABLE->CLAIMED': 'Warm pool allocation claimed by workflow',
  'CLAIMED->ACTIVE': 'Agent reports sync_complete (files downloaded successfully)',
  'CLAIMED->FAILED': 'Agent reports sync failure or timeout',
  'ACTIVE->COMPLETE': 'Run completes (success or failure); sets debug_hold_until if configured; replenishes warm pool',
  'ACTIVE->FAILED': 'Lifecycle error during execution',
};

// ACTIVE->FAILED concrete triggers:
// - Agent heartbeat timeout (exceeds STALE_DETECTION_MS, see timing.ts.l2.txt)
//   Constraint: HEARTBEAT_INTERVAL < HEARTBEAT_ACK_TIMEOUT <= STALE_DETECTION
// - SSH connection loss to instance (network partition)
// - Instance termination by provider (spot interruption, maintenance)
// - Fatal agent error (agent crash, OOM, unrecoverable error)
```

## StateValidation

```typescript
function validateAllocationTransition(
  currentStatus: AllocationStatus,
  newStatus: AllocationStatus
): { valid: boolean; error?: string } {
  const allowed = VALID_TRANSITIONS[currentStatus];

  if (!allowed.includes(newStatus)) {
    return {
      valid: false,
      error: `Invalid allocation transition: ${currentStatus} -> ${newStatus}`,
    };
  }

  return { valid: true };
}

function allowsSSHAccess(allocation: Allocation): boolean {
  if (allocation.status === 'ACTIVE') return true;
  if (allocation.status === 'COMPLETE' && allocation.debug_hold_until) {
    return allocation.debug_hold_until > Date.now();
  }
  return false;
}

function isInWarmPool(status: AllocationStatus): boolean {
  return status === 'AVAILABLE';
}
```

## RaceConditionActiveComplete

Both handlers use `UPDATE ... WHERE status = 'ACTIVE'` (first-write-wins CAS).
Only one succeeds; loser sees `changes === 0` and logs warning.

Key properties:
- Workflow checks `workflow_events` for `run_complete`; if found, treats as success even if allocation FAILED
- Agent wins race -> COMPLETE; timeout handler sees COMPLETE, returns
- Timeout wins race -> FAILED; workflow finds run_complete event, treats as success

### createAvailableAllocation

```typescript
async function createAvailableAllocation(
  instanceId: string,
  options: { user?: string; workdir?: string; manifestId?: string } = {}
): Promise<Allocation> {
  const instance = await db.getInstanceRecord(instanceId);

  // Determine workdir (use counter for isolation)
  const existingCount = await db.query(
    'SELECT COUNT(*) as count FROM allocations WHERE instance_id = ?',
    [instanceId]
  );
  const allocNumber = existingCount[0].count + 1;
  const workdir = options.workdir ?? `/home/${options.user ?? 'ubuntu'}/work_${allocNumber}`;

  // Insert allocation record - ID auto-generated by database (INTEGER PRIMARY KEY AUTOINCREMENT)
  const result = await db.execute(`
    INSERT INTO allocations (
      run_id, instance_id, status, user, workdir, current_manifest_id,
      debug_hold_until, created_at, updated_at
    ) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)
  `, [
    null, // run_id is NULL for AVAILABLE
    instanceId,
    'AVAILABLE',
    options.user ?? getDefaultUser(instance.provider),
    workdir,
    options.manifestId ?? null,
    null, // debug_hold_until
    Date.now(),
    Date.now(),
  ]);

  const allocationId = result.lastInsertRowid;
  await prepareWorkdir(instanceId, workdir, options.user ?? getDefaultUser(instance.provider));

  return materializeAllocation(allocationId);
}
```

### queryWarmPool (with scoring)

```typescript
interface WarmPoolQuery {
  provider?: string;
  spec?: string;
  initChecksum?: string;
  region?: string;
  excludeInstanceIds?: string[];
}

interface WarmPoolMatch extends Allocation {
  spec: string;
  init_checksum: string | null;
  region: string;
  checksum_score: number;
}

async function queryWarmPool(query: WarmPoolQuery, limit = 10): Promise<WarmPoolMatch[]> {
  const sql = `
    SELECT a.*, i.spec, i.init_checksum, i.region,
      CASE
        WHEN i.init_checksum = ? THEN 100  -- Exact checksum match
        WHEN i.init_checksum IS NULL THEN 50  -- Vanilla instance
        ELSE 0  -- Incompatible
      END as checksum_score
    FROM allocations a
    JOIN instances i ON a.instance_id = i.id
    WHERE a.status = 'AVAILABLE'
      AND i.workflow_state LIKE '%:complete'
      AND i.current_manifest_id IS NULL
      AND (? IS NULL OR i.provider = ?)
      AND (? IS NULL OR i.spec = ?)
      AND (i.init_checksum IS NULL OR i.init_checksum = ? OR ? IS NULL)
      AND i.last_heartbeat > ?
    ORDER BY checksum_score DESC, a.created_at ASC
    LIMIT ?
  `;

  return db.query(sql, [
    query.initChecksum, query.provider, query.provider,
    query.spec, query.spec, query.initChecksum, query.initChecksum,
    Date.now() - TIMING.STALE_DETECTION_MS, limit
  ]);
}
```

### claimWarmPoolAllocation (atomic with retry)

```typescript
async function claimWarmPoolAllocation(
  query: WarmPoolQuery,
  runId: string,
  maxRetries = 3
): Promise<{ success: boolean; allocation?: Allocation; error?: string }> {
  for (let attempt = 0; attempt < maxRetries; attempt++) {
    const candidates = await queryWarmPool(query, 5);
    if (candidates.length === 0) {
      return { success: false, error: 'No warm pool allocations match criteria' };
    }

    for (const candidate of candidates) {
      const result = await db.transaction(async () => {
        const alloc = await db.getAllocationForUpdate(candidate.id);

        if (!alloc || alloc.status !== 'AVAILABLE' || alloc.run_id !== null) {
          return { success: false, error: 'Allocation no longer available' };
        }

        await db.updateAllocation(candidate.id, {
          status: 'CLAIMED',
          run_id: runId,
          updated_at: Date.now(),
        });

        return { success: true, allocation: await materializeAllocation(candidate.id) };
      });

      if (result.success) return result;
    }

    await sleep(50 * (attempt + 1)); // Exponential backoff
  }

  return { success: false, error: 'Failed to claim after max retries' };
}
```

## ActiveCompleteTransition

```typescript
async function transitionToComplete(
  allocationId: string,
  options: {
    exitCode: number;
    keepOnComplete?: boolean;
    debugHoldDurationMs?: number;
  }
): Promise<void> {
  await db.transaction(async () => {
    const allocation = await db.getAllocationForUpdate(allocationId);
    if (!allocation || allocation.status !== 'ACTIVE') {
      throw new Error(`Allocation ${allocationId} not in ACTIVE state`);
    }

    const now = Date.now();
    const updates: Partial<Allocation> = {
      status: 'COMPLETE',
      updated_at: now,
    };

    // Set debug_hold_until if configured
    if (options.keepOnComplete) {
      const holdDurationMs = options.debugHoldDurationMs ??
        (options.exitCode === 0
          ? TIMING.DEFAULT_HOLD_MS
          : TIMING.DEFAULT_HOLD_MS * 3);  // Longer hold for failures

      updates.debug_hold_until = now + holdDurationMs;
    }

    await db.updateAllocation(allocationId, updates);

    // Warm pool replenishment: create new AVAILABLE allocation if eligible
    const instance = await materializeInstance(allocation.instance_id);
    const shouldReplenish = await checkReplenishmentEligibility(instance, allocation);

    if (shouldReplenish) {
      await createAvailableAllocation(instance.id, {
        user: allocation.user,
        manifestId: allocation.current_manifest_id,
      });
    }
  });
}

async function checkReplenishmentEligibility(
  instance: Instance,
  completedAllocation: Allocation
): Promise<boolean> {
  // Only replenish if instance is healthy and not scheduled for termination
  if (instance.workflow_state?.includes('terminate')) return false;
  if (instance.last_heartbeat < Date.now() - TIMING.STALE_DETECTION_MS) return false;

  // Check if instance is at capacity (MAX_ALLOCATIONS_PER_INSTANCE)
  const activeCount = await db.query(`
    SELECT COUNT(*) as count FROM allocations
    WHERE instance_id = ? AND status IN ('CLAIMED', 'ACTIVE')
  `, [instance.id]);

  if (activeCount[0].count >= MAX_ALLOCATIONS_PER_INSTANCE) return false;

  return true;
}
```

## DebugHoldExtension

Called from heartbeat handler when agent reports active SSH sessions.

```typescript
async function extendAllocationDebugHold(
  allocationId: string,
  options: { extensionMs: number; maxExtensionMs?: number }
): Promise<{ success: boolean; newDebugHoldUntil?: number }> {
  return db.transaction(async () => {
    const allocation = await db.getAllocationForUpdate(allocationId);

    // Only extend if ACTIVE or (COMPLETE with existing debug_hold_until)
    if (!allocation ||
        (allocation.status !== 'ACTIVE' &&
         !(allocation.status === 'COMPLETE' && allocation.debug_hold_until))) {
      return { success: false };
    }

    const now = Date.now();
    const currentDebugHoldUntil = allocation.debug_hold_until ?? now;

    // Calculate when allocation completed (use updated_at as proxy)
    const completedAt = allocation.updated_at;

    let newDebugHoldUntil = Math.max(currentDebugHoldUntil, now) + options.extensionMs;
    const maxHoldUntil = completedAt + (options.maxExtensionMs ?? TIMING.MAX_SSH_EXTENSION_MS);

    if (newDebugHoldUntil > maxHoldUntil) newDebugHoldUntil = maxHoldUntil;

    // Absolute hard limit (24 hours)
    const absoluteLimit = completedAt + TIMING.ABSOLUTE_MAX_HOLD_MS;
    if (newDebugHoldUntil > absoluteLimit) newDebugHoldUntil = absoluteLimit;

    await db.updateAllocation(allocationId, {
      debug_hold_until: newDebugHoldUntil,
      updated_at: Date.now(),
    });

    return { success: true, newDebugHoldUntil };
  });
}
```

## TerminationBlocking

```typescript
async function canTerminateInstance(instanceId: string): Promise<boolean> {
  const now = Date.now();

  // Check if any allocation on this instance has an active debug hold
  const blockedAllocations = await db.query(`
    SELECT id FROM allocations
    WHERE instance_id = ?
      AND status = 'COMPLETE'
      AND debug_hold_until IS NOT NULL
      AND debug_hold_until > ?
    LIMIT 1
  `, [instanceId, now]);

  return blockedAllocations.length === 0;
}

async function terminateInstance(instanceId: string): Promise<void> {
  const canTerminate = await canTerminateInstance(instanceId);

  if (!canTerminate) {
    throw new Error(
      `Cannot terminate instance ${instanceId}: has allocations with active debug holds. ` +
      `Use 'repl release <allocation-id>' to clear holds or wait for expiry.`
    );
  }

  // Proceed with termination workflow
  await launchTerminateIntent(instanceId);
}
```

async function reconcileClaimedAllocations(): Promise<void> {
  const staleAllocations = await db.query(`
    SELECT id FROM allocations
    WHERE status = 'CLAIMED'
      AND updated_at < ?
  `, [Date.now() - TIMING.CLAIMED_TIMEOUT_MS]);

  for (const allocation of staleAllocations) {
    await db.updateAllocation(allocation.id, {
      status: 'FAILED',
      updated_at: Date.now()
    });

    logger.warn('CLAIMED allocation timed out', {
      allocation_id: allocation.id,
      age_ms: Date.now() - allocation.updated_at
    });
  }
}

## HoldExpiryBackgroundTask

Clears expired debug_hold_until timestamps. Does NOT transition allocation state.
See `../background/hold-expiry.ts.l1.txt` for task scheduling details.

```typescript
async function holdExpiryTask(): Promise<void> {
  const now = Date.now();

  // Find allocations with expired debug holds
  const expiredAllocations = await db.query(`
    SELECT id, instance_id, debug_hold_until
    FROM allocations
    WHERE status = 'COMPLETE'
      AND debug_hold_until IS NOT NULL
      AND debug_hold_until <= ?
  `, [now]);

  for (const alloc of expiredAllocations) {
    await db.updateAllocation(alloc.id, {
      debug_hold_until: null,
      updated_at: now,
    });

    // Trigger SSH config regeneration (allocation no longer SSH-accessible)
    await triggerSSHConfigRegeneration();
  }

  if (expiredAllocations.length > 0) {
    logger.info(`Cleared ${expiredAllocations.length} expired debug holds`);
  }
}

// Start background task
setInterval(holdExpiryTask, TIMING.HOLD_EXPIRY_CHECK_INTERVAL_MS);
```

## TailscaleEnsureEndpoint

Tailscale state is stored as TailscaleMachine objects in the objects table (see Ch03),
NOT as columns on the instances table. Queries use object_tags to find the
tailscale_machine object for a given instance.

```typescript
// POST /api/v1/instances/:id/tailscale-ensure
// Single endpoint that queries status and triggers installation if needed
interface TailscaleStatusResponse {
  status: TailscaleStatus;
  ip?: string;  // Tailscale IP when status is 'ready'
}

// Helper: get Tailscale state from objects table
async function getTailscaleState(instanceId: string): Promise<{ status: TailscaleStatus; ip: string | null }> {
  const row = await db.queryOne(`
    SELECT o.metadata_json
    FROM objects o
    JOIN object_tags ot ON o.id = ot.object_id
    WHERE o.type = 'tailscale_machine'
      AND ot.tag = ?
  `, [`instance_id:${instanceId}`]);

  if (!row) {
    return { status: 'not_installed', ip: null };
  }

  const metadata = JSON.parse(row.metadata_json);
  return {
    status: metadata.tailscale_status ?? 'not_installed',
    ip: metadata.tailscale_ip ?? null,
  };
}

// Helper: update Tailscale state in objects table
async function updateTailscaleStatus(instanceId: string, status: TailscaleStatus): Promise<void> {
  const existing = await db.queryOne(`
    SELECT o.id, o.metadata_json
    FROM objects o
    JOIN object_tags ot ON o.id = ot.object_id
    WHERE o.type = 'tailscale_machine'
      AND ot.tag = ?
  `, [`instance_id:${instanceId}`]);

  if (existing) {
    const metadata = JSON.parse(existing.metadata_json);
    metadata.tailscale_status = status;
    await db.execute(`UPDATE objects SET metadata_json = ?, updated_at = ? WHERE id = ?`,
      [JSON.stringify(metadata), Date.now(), existing.id]);
  } else {
    // Create new tailscale_machine object
    const objectId = await db.insertObject({
      type: 'tailscale_machine',
      metadata_json: JSON.stringify({ tailscale_status: status, tailscale_ip: null }),
    });
    await db.execute(`INSERT INTO object_tags (object_id, tag) VALUES (?, ?)`,
      [objectId, `instance_id:${instanceId}`]);
  }
}

app.post('/api/v1/instances/:id/tailscale-ensure', async (req, res) => {
  const instanceId = req.params.id;
  const tsState = await getTailscaleState(instanceId);

  // Return current status if already installing or ready
  if (tsState.status === 'installing' || tsState.status === 'ready') {
    return res.json({
      status: tsState.status,
      ip: tsState.ip ?? undefined,
    });
  }

  // If not installed, trigger installation
  if (!tsState.status || tsState.status === 'not_installed') {
    // Update status to 'installing' in objects table
    await updateTailscaleStatus(instanceId, 'installing');

    // Enqueue message to instance agent
    await enqueueInstanceMessage(instanceId, {
      type: 'trigger_tailscale',
      timestamp: Date.now(),
    });

    return res.json({ status: 'installing' });
  }

  // Handle 'failed' status - return as-is but allow retry if explicitly requested
  res.json({
    status: tsState.status,
    ip: tsState.ip ?? undefined,
  });
});
```
