## ProviderError Base Class

```typescript
/**
 * Base class for all provider errors.
 * Provides structured error information for handling.
 */
abstract class ProviderError extends Error {
  abstract readonly code: ProviderErrorCode;
  abstract readonly category: ProviderErrorCategory;
  abstract readonly retryable: boolean;
  abstract readonly retryAfterMs?: number;

  constructor(
    message: string,
    public readonly provider: ProviderName,
    public readonly details?: Record<string, unknown>
  ) {
    super(message);
    this.name = this.constructor.name;
  }
}

type ProviderErrorCode =
  | 'CAPACITY_ERROR'
  | 'AUTH_ERROR'
  | 'RATE_LIMIT_ERROR'
  | 'SPOT_INTERRUPTED'
  | 'QUOTA_EXCEEDED'
  | 'INVALID_SPEC'
  | 'REGION_UNAVAILABLE'
  | 'NETWORK_ERROR'
  | 'TIMEOUT_ERROR'
  | 'NOT_FOUND'
  | 'ALREADY_EXISTS'
  | 'INVALID_STATE'
  | 'PROVIDER_INTERNAL'
  | 'UNSUPPORTED_OPERATION';

type ProviderErrorCategory =
  | 'capacity'     // Resource availability issues
  | 'auth'         // Authentication/authorization
  | 'rate_limit'   // Rate limiting / throttling
  | 'validation'   // Invalid input
  | 'not_found'    // Resource not found
  | 'conflict'     // State conflict
  | 'internal';    // Provider internal error
```

## Specific Error Classes

```typescript
/**
 * Insufficient capacity to spawn instance.
 * May suggest alternative regions/specs.
 */
class CapacityError extends ProviderError {
  readonly code = 'CAPACITY_ERROR' as const;
  readonly category = 'capacity' as const;
  readonly retryable = true;

  constructor(
    provider: ProviderName,
    public readonly spec: string,
    public readonly region: string,
    public readonly alternativeRegions?: string[],
    public readonly alternativeSpecs?: string[],
    public readonly retryAfterMs?: number
  ) {
    super(
      `Insufficient capacity for ${spec} in ${region}`,
      provider,
      { spec, region, alternativeRegions, alternativeSpecs }
    );
  }
}

/**
 * Authentication or authorization failure.
 * Credentials invalid or expired.
 */
class AuthError extends ProviderError {
  readonly code = 'AUTH_ERROR' as const;
  readonly category = 'auth' as const;
  readonly retryable = false;

  constructor(
    provider: ProviderName,
    public readonly reason: 'invalid_credentials' | 'expired' | 'insufficient_permissions',
    message?: string
  ) {
    super(
      message ?? `Authentication failed: ${reason}`,
      provider,
      { reason }
    );
  }
}

/**
 * Rate limited by provider.
 * Includes retry-after information.
 */
class RateLimitError extends ProviderError {
  readonly code = 'RATE_LIMIT_ERROR' as const;
  readonly category = 'rate_limit' as const;
  readonly retryable = true;

  constructor(
    provider: ProviderName,
    public readonly retryAfterMs: number,
    public readonly limitType?: 'api' | 'resource' | 'budget'
  ) {
    super(
      `Rate limited by ${provider}, retry after ${retryAfterMs}ms`,
      provider,
      { retryAfterMs, limitType }
    );
  }
}

/**
 * Spot instance interrupted by provider.
 * Contains interruption details and timing.
 */
class SpotInterrupted extends ProviderError {
  readonly code = 'SPOT_INTERRUPTED' as const;
  readonly category = 'capacity' as const;
  readonly retryable = true;

  constructor(
    provider: ProviderName,
    public readonly instanceId: string,
    public readonly interruptionTime: number,
    public readonly warningTimeMs?: number,  // Time between warning and interruption
    public readonly retryAfterMs?: number
  ) {
    super(
      `Spot instance ${instanceId} interrupted`,
      provider,
      { instanceId, interruptionTime, warningTimeMs }
    );
  }
}
```

## Error Handling Helpers

```typescript
/**
 * Map provider-specific errors to our error taxonomy.
 * Fallback for unmapped errors - each provider implements its own mapping.
 */
function mapProviderError(
  provider: ProviderName,
  error: unknown
): ProviderError {
  if (error instanceof ProviderError) return error;

  const message = error instanceof Error ? error.message : String(error);
  return new class extends ProviderError {
    readonly code = 'PROVIDER_INTERNAL' as const;
    readonly category = 'internal' as const;
    readonly retryable = false;
  }(message, provider, { originalError: error });
}

/**
 * Determine if error should trigger Retry-with-Alternative pattern.
 */
function shouldRetryWithAlternative(error: ProviderError): boolean {
  return (
    error.code === 'CAPACITY_ERROR' ||
    error.code === 'SPOT_INTERRUPTED' ||
    error.code === 'REGION_UNAVAILABLE'
  );
}

/**
 * Determine if error should trigger simple retry.
 */
function shouldRetry(error: ProviderError): boolean {
  return error.retryable && error.category !== 'auth';
}
```
