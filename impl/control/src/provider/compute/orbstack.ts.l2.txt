## OrbStack Bootstrap Generation

```typescript
/**
 * OrbStack uses shell script, not cloud-init.
 * Bootstrap is executed via `orb run` after VM creation.
 */
class OrbStackProvider implements Provider<OrbStackInstance> {
  generateBootstrap(config: BootstrapConfig): BootstrapScript {
    const script = `#!/bin/bash
set -e

# Configure agent
mkdir -p /etc/skyrepl
cat > /etc/skyrepl/config.json << 'EOF'
${JSON.stringify({
  controlPlaneUrl: config.controlPlaneUrl,
  registrationToken: config.registrationToken,
})}
EOF

# Download agent
curl -fsSL ${config.agentUrl} -o /usr/local/bin/skyrepl-agent
chmod +x /usr/local/bin/skyrepl-agent

${config.initScript ?? ''}

# Start agent
/usr/local/bin/skyrepl-agent start
`;

    const checksum = crypto.createHash('sha256').update(script).digest('hex');

    return {
      content: script,
      format: 'shell',
      checksum,
    };
  }
}
```
