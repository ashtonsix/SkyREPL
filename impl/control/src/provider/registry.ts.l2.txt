## Compute Provider Registry

```typescript
/**
 * Provider registry for runtime and test discovery.
 * Shared between production code and test suite.
 *
 * Providers are organized into:
 * - compute/: Cloud compute providers (AWS, Lambda Labs, RunPod, OrbStack)
 * - feature/: Cross-cutting feature providers (Tailscale, etc.)
 */

// Lazy-load providers to avoid circular dependencies and startup overhead
export const providerRegistry = {
  aws: () => import('./compute/aws'),
  lambda: () => import('./compute/lambda'),
  runpod: () => import('./compute/runpod'),
  orbstack: () => import('./compute/orbstack'),
} as const;

export type ProviderName = keyof typeof providerRegistry;

// Provider cache to avoid repeated imports
const providerCache = new Map<ProviderName, Provider>();
```

## Runtime Access

```typescript
/**
 * Get a provider instance by name.
 * Lazy-loads and caches provider on first access.
 */
export async function getProvider(name: ProviderName): Promise<Provider> {
  if (providerCache.has(name)) {
    return providerCache.get(name)!;
  }

  const loader = providerRegistry[name];
  if (!loader) {
    throw new Error(`Provider '${name}' not found in registry`);
  }

  const module = await loader();
  const provider = module.default;

  providerCache.set(name, provider);
  return provider;
}

/**
 * Get list of all registered provider names.
 */
export function getAllProviders(): ProviderName[] {
  return Object.keys(providerRegistry) as ProviderName[];
}

/**
 * Check if provider is registered.
 */
export function isProviderRegistered(name: string): name is ProviderName {
  return name in providerRegistry;
}

/**
 * Clear provider cache (useful for testing).
 */
export function clearProviderCache(): void {
  providerCache.clear();
}
```

## Dynamic Provider Registration

```typescript
/**
 * Complete provider registration including hooks and workflows.
 */
interface ProviderRegistration {
  /** Provider implementation */
  provider: Provider;

  /** Lifecycle hooks */
  hooks?: ProviderLifecycleHooks;

  /** Workflow contracts */
  contracts?: ProviderWorkflowContract[];
}

// Dynamic registry for plugins (future extensibility)
const dynamicProviders = new Map<string, ProviderRegistration>();

/**
 * Register a provider with the control plane.
 * Includes provider instance, hooks, and contracts.
 */
function registerProvider(registration: ProviderRegistration): void {
  const { provider, hooks, contracts } = registration;

  // Register provider instance
  providers.set(provider.name, provider);

  // Register hooks
  if (hooks) {
    providerHooks.set(provider.name, hooks);
  }

  // Register contracts
  for (const contract of contracts ?? []) {
    providerWorkflowContracts.set(`${provider.name}:${contract.type}`, contract);
  }

  debug('PROVIDER/register', {
    provider: provider.name,
    capabilities: provider.capabilities,
    hasHooks: !!hooks,
  });
}

/**
 * Unregister a dynamic provider.
 */
export function unregisterProvider(name: string): void {
  dynamicProviders.delete(name);
  providerCache.delete(name as ProviderName);
}
```

## Test Utilities

```typescript
/**
 * Get list of providers suitable for testing.
 * OrbStack always included (local, no credentials).
 * Cloud providers gated by credential availability.
 */
export function getTestableProviders(): ProviderName[] {
  const all = getAllProviders();
  const testable: ProviderName[] = ['orbstack'];

  for (const name of all) {
    if (name === 'orbstack') continue;
    switch (name) {
      case 'aws':
        if (process.env.AWS_ACCESS_KEY_ID) testable.push('aws');
        break;
      case 'lambda':
        if (process.env.LAMBDA_API_KEY) testable.push('lambda');
        break;
      case 'runpod':
        if (process.env.RUNPOD_API_KEY) testable.push('runpod');
        break;
    }
  }

  // Allow explicit provider selection via env var
  if (process.env.TEST_PROVIDERS) {
    const explicit = process.env.TEST_PROVIDERS.split(',') as ProviderName[];
    return explicit.filter((name) => all.includes(name));
  }

  return testable;
}

/**
 * Get list of feature providers suitable for testing.
 */
export function getTestableFeatureProviders(): FeatureProviderName[] {
  const all = getAllFeatureProviders();
  const testable: FeatureProviderName[] = [];

  for (const name of all) {
    switch (name) {
      case 'tailscale':
        if (process.env.TAILSCALE_API_KEY || process.env.TAILSCALE_MOCK) {
          testable.push('tailscale');
        }
        break;
    }
  }

  if (process.env.TEST_FEATURE_PROVIDERS) {
    const explicit = process.env.TEST_FEATURE_PROVIDERS.split(',') as FeatureProviderName[];
    return explicit.filter((name) => all.includes(name));
  }

  return testable;
}
```

## Contract Testing Pattern (Vitest)

```typescript
/**
 * Contract compliance tests.
 * Run same suite against all providers using describe.each.
 */

// tests/control/unit/provider-contracts.test.ts
import { describe, it, expect, beforeAll } from 'vitest';
import { providerRegistry, ProviderName } from '../../../control/src/provider/registry';
import type { Provider } from '../../../control/src/provider/types';

const providers = Object.keys(providerRegistry) as ProviderName[];

describe.each(providers)('Provider %s', (name) => {
  let provider: Provider;

  beforeAll(async () => {
    const mod = await providerRegistry[name]();
    provider = mod.default;
  });

  describe('SpawnInstanceContract compliance', () => {
    it('returns required fields on success', async () => {
      const result = await provider.spawn(testOptions);
      expect(result).toHaveProperty('id');
      expect(result).toHaveProperty('status');
      expect(result.status).toMatch(/pending|starting|running/);
    });

    it('is idempotent when capability declared', async () => {
      if (!provider.capabilities.idempotentSpawn) return;

      const opts = { ...testOptions, clientRequestId: 'idem-1' };
      const r1 = await provider.spawn(opts);
      const r2 = await provider.spawn(opts);
      expect(r1.id).toBe(r2.id);
    });

    it('emits required tags', async () => {
      const result = await provider.spawn({
        ...testOptions,
        tags: { 'skyrepl:test': 'true' },
      });
      const instance = await provider.get(result.id);
      expect(instance?.metadata?.tags).toHaveProperty('skyrepl:test', 'true');
    });
  });

  describe('TerminateInstanceContract compliance', () => {
    it('handles non-existent instance gracefully', async () => {
      await expect(provider.terminate('nonexistent'))
        .rejects.toThrow(NotFoundError);
    });

    it('is idempotent (terminate twice = same result)', async () => {
      const instance = await provider.spawn(testOptions);
      await provider.terminate(instance.id);
      await expect(provider.terminate(instance.id))
        .resolves.not.toThrow();
    });
  });

  describe('ListFilter compliance', () => {
    it('filters by region', async () => {
      const results = await provider.list({ region: 'us-east-1' });
      expect(results.every(i => i.region?.startsWith('us-east-1'))).toBe(true);
    });

    it('filters by status', async () => {
      const results = await provider.list({ status: ['running', 'pending'] });
      expect(results.every(i => ['running', 'pending'].includes(i.status))).toBe(true);
    });
  });

  describe('BootstrapScript compliance', () => {
    it('generates valid bootstrap script', async () => {
      const script = provider.generateBootstrap(testBootstrapConfig);
      expect(script.content).toBeTruthy();
      expect(script.format).toMatch(/cloud-init|shell|ignition/);
      expect(script.checksum).toHaveLength(64); // SHA-256
    });

    it('includes agent download in bootstrap', async () => {
      const script = provider.generateBootstrap(testBootstrapConfig);
      expect(script.content).toContain(testBootstrapConfig.agentUrl);
    });
  });

  // Capability-gated tests
  describe('Snapshot capability', () => {
    it('creates snapshot when supported', async () => {
      if (!provider.capabilities.snapshots) return;

      const instance = await provider.spawn(testOptions);
      const request = await provider.createSnapshot!(instance.id, {
        name: 'test-snapshot',
      });

      expect(request).toHaveProperty('requestId');
      expect(request.status.status).toMatch(/pending|creating/);
    });
  });

  describe('Spot capability', () => {
    it('requests spot instance when supported', async () => {
      if (!provider.capabilities.spot) return;

      const request = await provider.requestSpotInstance!({
        ...testOptions,
        maxSpotPrice: 0.50,
      });

      expect(request).toHaveProperty('requestId');
      expect(request.status).toMatch(/pending|active|fulfilled/);
    });
  });
});
```
