## ProviderInterface

```typescript
/**
 * Provider interface defines the contract all providers must implement.
 * Core methods are required; optional capabilities are gated by flags.
 *
 * CONTRACTS: Provider contracts are enforced via TypeScript types + unit tests.
 * Every provider MUST pass interface compliance tests that verify correct behavior
 * (spawn returns required fields, terminate handles non-existent instances, etc.).
 * This is not runtime contract checking in production, just good tests that ensure
 * providers satisfy the logical contract.
 */
interface Provider<
  TInstance extends ProviderInstance = ProviderInstance,
  TSnapshot extends ProviderSnapshot = ProviderSnapshot
> {
  /** Provider identifier (aws, lambda, runpod, orbstack) */
  readonly name: ProviderName;

  /** Capability flags for optional features */
  readonly capabilities: ProviderCapabilities;

  // ============================================================
  // CORE METHODS (Required)
  // ============================================================

  /**
   * Spawn a new instance with the given configuration.
   * @returns Provider-specific instance with at minimum { id, status }
   */
  spawn(options: SpawnOptions): Promise<TInstance>;

  /**
   * Terminate an instance.
   * @param providerId - Provider's instance ID (i-xxx, etc.)
   * @throws ProviderError if instance not found or already terminated
   */
  terminate(providerId: string): Promise<void>;

  /**
   * List instances, optionally filtered by status.
   * Used for reconciliation and drift detection.
   */
  list(filter?: ListFilter): Promise<TInstance[]>;

  /**
   * Get a single instance by provider ID.
   * @returns Instance or null if not found
   */
  get(providerId: string): Promise<TInstance | null>;

  /**
   * Generate cloud-init/bootstrap script for instance initialization.
   * Encapsulates all provider-specific boot configuration.
   */
  generateBootstrap(config: BootstrapConfig): BootstrapScript;

  // ============================================================
  // OPTIONAL METHODS (Gated by capabilities)
  // ============================================================

  /**
   * Create a snapshot of an instance.
   * Only available if capabilities.snapshots === true
   */
  createSnapshot?(
    providerId: string,
    options: CreateSnapshotOptions
  ): Promise<SnapshotRequest>;

  /**
   * Get snapshot creation status.
   * Only available if capabilities.snapshots === true
   */
  getSnapshotStatus?(requestId: string): Promise<SnapshotStatus>;

  /**
   * Delete a snapshot.
   * Only available if capabilities.snapshots === true
   */
  deleteSnapshot?(providerSnapshotId: string): Promise<void>;

  /**
   * Get snapshot by name.
   * Only available if capabilities.snapshots === true
   */
  getSnapshotByName?(name: string): Promise<TSnapshot | null>;

  /**
   * Request spot instance.
   * Only available if capabilities.spot === true
   */
  requestSpotInstance?(options: SpotOptions): Promise<SpotRequest>;

  /**
   * Cancel spot request.
   * Only available if capabilities.spot === true
   */
  cancelSpotRequest?(requestId: string): Promise<void>;

  /**
   * Get current spot prices.
   * Only available if capabilities.spot === true
   */
  getSpotPrices?(spec: string, regions?: string[]): Promise<SpotPriceInfo[]>;
}

// Provider names as const union
type ProviderName = 'aws' | 'lambda' | 'runpod' | 'orbstack';
```

interface SpawnOptions {
  /** Instance specification (g5.xlarge, A100x1, etc.) */
  spec: string;

  /** Target region/zone */
  region?: string;

  /** Use spot/preemptible pricing (requires capabilities.spot) */
  spot?: boolean;

  /** Maximum spot price ($/hour) */
  maxSpotPrice?: number;

  /** Init script checksum for warm pool matching */
  initChecksum?: string;

  /** Tags to apply to instance */
  tags?: Record<string, string>;

  /** Bootstrap configuration */
  bootstrap: BootstrapConfig;

  /** Spawn from snapshot (requires capabilities.snapshots) */
  snapshotId?: string;

  /** Attach existing volume (requires capabilities.warmVolumes) */
  volumeId?: string;

  /** Subnet/network configuration (requires capabilities.customNetworking) */
  networkConfig?: NetworkConfig;

  /** Client-provided request ID for idempotency */
  clientRequestId?: string;
}

interface ProviderInstance {
  /** Provider's instance ID (i-xxx, vm-xxx, etc.) */
  id: string;

  /** Current status in provider's system */
  status: ProviderInstanceStatus;

  /** Instance specification */
  spec: string;

  /** Region/zone */
  region?: string;

  /** Public IP address (null if not assigned yet) */
  ip?: string | null;

  /** Private IP address */
  privateIp?: string | null;

  /** Creation timestamp */
  createdAt: number;

  /** Is this a spot/preemptible instance */
  isSpot: boolean;

  /** Spot request ID (if spot instance) */
  spotRequestId?: string;

  /** Provider-specific metadata (extended by concrete types) */
  metadata?: Record<string, unknown>;
}

type ProviderInstanceStatus =
  | 'pending'      // Request accepted, not yet running
  | 'starting'     // Instance booting
  | 'running'      // Instance running and accessible
  | 'stopping'     // Instance stopping
  | 'stopped'      // Instance stopped (hibernated)
  | 'terminating'  // Instance being terminated
  | 'terminated'   // Instance terminated
  | 'error';       // Instance in error state

## FeatureProviderInterface

```typescript
/**
 * Feature providers attach to existing instances rather than creating compute.
 * They enable cross-cutting concerns like networking, logging, and development tools.
 */
interface FeatureProvider<TConfig, TState> {
  /** Feature provider name (tailscale, wandb, vscode-server) */
  name: string;

  /** Feature capabilities */
  capabilities: FeatureCapabilities;

  /**
   * Attach feature to an instance.
   * Called during instance spawn or hot-attach after running.
   */
  attach(instanceId: string, config: TConfig): Promise<TState>;

  /**
   * Detach feature from instance.
   * Called during instance termination or feature removal.
   */
  detach(instanceId: string): Promise<void>;

  /**
   * Get current feature state for an instance.
   * Returns null if feature not attached or state unavailable.
   */
  status(instanceId: string): Promise<TState | null>;

  /**
   * Optional: Reconcile feature state during orphan detection.
   * Called during orphan detection cycle to find feature-specific orphans.
   */
  reconcile?(context: FeatureReconcileContext): Promise<ReconcileResult>;

  /**
   * Optional: Called on instance heartbeat for feature maintenance.
   */
  onInstanceHeartbeat?(instanceId: string): Promise<void>;
}

interface FeatureCapabilities {
  /** Feature requires bootstrap script injection */
  requiresBootstrap: boolean;

  /** Feature can attach after instance already running */
  supportsHotAttach: boolean;

  /** Feature can detach without terminating instance */
  supportsDetach: boolean;

  /** Feature works with all compute providers */
  crossProvider: boolean;
}

interface FeatureReconcileContext {
  /** Instances that should have this feature attached */
  expectedAttachments: Array<{
    instanceId: string;
    config: unknown;
  }>;

  /** Current feature state from provider */
  actualState: Map<string, unknown>;
}
```

interface SpawnFeatureConfig {
  provider: FeatureProviderName;

// Bootstrap generation includes feature setup
function generateBootstrap(config: BootstrapConfig, features?: BootstrapFeature[]): BootstrapScript {
  const commands: string[] = [
    // Core agent setup ...
  ];

  // Feature-specific setup
  for (const feature of features ?? []) {
    const fp = getFeatureProvider(feature.name);
    if (fp.capabilities.requiresBootstrap) {
      commands.push(...fp.generateBootstrapCommands(feature.config));
    }
  }
  // ...
}

// Instance termination includes feature cleanup
async function terminateInstance(instance: Instance): Promise<void> {
  // Cleanup features before terminating compute
  for (const feature of instance.attachedFeatures) {
    const fp = getFeatureProvider(feature.provider);
    await fp.detach(instance.id);
  }

  // Terminate compute resource
  const provider = getProvider(instance.provider);
  await provider.terminate(instance.providerId);
}
