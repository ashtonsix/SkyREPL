## InMemoryCache

```typescript
/**
 * Providers cache ephemeral data (spot prices, capacity info) in memory with TTL.
 * Not persisted - re-fetched on control plane restart.
 */

async function getWithCache<T>(
  key: string,
  ttlMs: number,
  fetcher: () => Promise<T>
): Promise<T> {
  const cached = cache.get(key);
  if (cached && Date.now() < cached.expiresAt) {
    return cached.value;
  }
  const value = await fetcher();
  cache.set(key, { value, expiresAt: Date.now() + ttlMs });
  return value;
}

// Usage
const spotPrices = await getWithCache(
  `aws:spot_prices:${region}:${spec}`,
  300_000, // 5 min TTL
  () => ec2.getSpotPrices(spec, region)
);
```

## LifecycleHooksInterface

```typescript
/**
 * Lifecycle hooks let providers respond to system events.
 * Core sends expectations, provider returns receipts.
 */

interface ProviderLifecycleHooks {
  /**
   * Called when control plane starts.
   * For initialization, cache warming, health checks.
   */
  onStartup?(): Promise<void>;

  /**
   * Called on graceful shutdown.
   * For cleanup, state persistence.
   */
  onShutdown?(): Promise<void>;

  /**
   * Unified heartbeat with expectations model.
   * Core sends what it needs, provider responds with what it did.
   */
  onHeartbeat?(expectations: HeartbeatExpectations): Promise<HeartbeatReceipts>;
}
```

## HeartbeatExpectationsModel

```typescript
/**
 * HeartbeatExpectations: Core tells provider which tasks to perform.
 */
interface HeartbeatExpectations {
  /** Tasks for provider to execute */
  tasks: HeartbeatTask[];

  /** Timestamp by which tasks should complete */
  deadline: number;
}

interface HeartbeatTask {
  /** Task type */
  type: 'reconcile' | 'health_check' | 'cache_refresh' | 'pool_maintenance';

  /** Task priority */
  priority: 'high' | 'normal' | 'low';

  /** Last time this task ran (for scheduling) */
  lastRun?: number;

  /** Task-specific parameters */
  params?: Record<string, unknown>;
}

/**
 * HeartbeatReceipts: Provider reports what it accomplished.
 */
interface HeartbeatReceipts {
  /** Receipts for requested tasks */
  receipts: TaskReceipt[];

  /** Provider-initiated tasks (not requested by core) */
  providerTasks?: ProviderTaskReport[];

  /**
   * Workflows launched for async fulfillment.
   *
   * Some tasks (like spot interruption handling or budget threshold responses)
   * require workflow orchestration rather than synchronous completion. When a
   * heartbeat task or external event triggers workflow creation, report it here.
   *
   * This enables the core to track workflows launched by providers without
   * requiring separate event channels. Most heartbeats leave this empty.
   */
  workflowsLaunched?: WorkflowLaunchReport[];

  /** Provider's preferred next heartbeat timing */
  nextHeartbeat?: {
    preferredInterval: number;
  };
}

interface WorkflowLaunchReport {
  /** Workflow ID returned from submission */
  workflowId: string;

  /** Workflow type (e.g., 'handle-spot-interruption', 'budget-cleanup') */
  workflowType: string;

  /** What triggered this workflow */
  trigger: 'spot_interruption' | 'budget_threshold' | 'drift_detected' | 'provider_event';

  /** Additional context */
  context?: Record<string, unknown>;
}

interface TaskReceipt {
  /** Task type (matches HeartbeatTask.type) */
  type: string;

  /** Task completion status */
  status: 'completed' | 'skipped' | 'deferred' | 'failed';

  /** Task result data */
  result?: unknown;

  /** Reason for skip/defer/failure */
  reason?: string;
}

interface ProviderTaskReport {
  /** Provider-specific task type */
  type: string;

  /** Task status */
  status: 'completed' | 'failed';

  /** Task result summary */
  result?: unknown;
}
```

## TTLConfiguration

```typescript
/**
 * TTL patterns for provider state management.
 */

interface TTLConfig {
  /** Default TTL for uncategorized entries */
  default: number;

  /** TTL by key prefix */
  byPrefix: Record<string, number>;

  /** Maximum allowed TTL */
  maxTtl: number;
}

const AWS_TTL_CONFIG: TTLConfig = {
  default: 300_000,  // 5 minutes
  maxTtl: 3600_000,  // 1 hour

  byPrefix: {
    'spot_prices:': 300_000,       // 5 minutes
    'capacity:': 60_000,           // 1 minute
    'instance:': 30_000,           // 30 seconds
    'warm_volume_pool:': null,     // No TTL (explicitly managed)
    'cost:': 3600_000,             // 1 hour
    'api_calls:': 60_000,          // 1 minute (rolling window)
  },
};

/**
 * Get appropriate TTL for a key.
 */
function getTTL(provider: ProviderName, key: string): number | null {
  const config = PROVIDER_TTL_CONFIGS[provider] ?? { default: 300_000, byPrefix: {}, maxTtl: 3600_000 };

  for (const [prefix, ttl] of Object.entries(config.byPrefix)) {
    if (key.startsWith(prefix)) {
      return ttl;
    }
  }

  return config.default;
}

/**
 * Wrapper that automatically applies TTL.
 */
async function setWithAutoTTL<T>(
  provider: ProviderName,
  key: string,
  value: T
): Promise<void> {
  const ttl = getTTL(provider, key);
  await providerState.set(provider, key, value, ttl ?? undefined);
}
```

## HookRegistry

```typescript
/**
 * Register and invoke provider lifecycle hooks.
 */

const providerHooks = new Map<ProviderName, ProviderLifecycleHooks>();

function registerProviderHooks(provider: ProviderName, hooks: ProviderLifecycleHooks): void {
  providerHooks.set(provider, hooks);
}

async function invokeHook(
  provider: ProviderName,
  hook: keyof ProviderLifecycleHooks,
  ...args: unknown[]
): Promise<void> {
  const hooks = providerHooks.get(provider);
  if (!hooks?.[hook]) return; // No-op if not registered

  try {
    await (hooks[hook] as Function)(...args);
  } catch (error) {
    // Hooks should not crash control plane
    debug(`HOOK/${provider}/${hook}/error`, { error });
  }
}

async function invokeAllHooks(
  hook: keyof ProviderLifecycleHooks,
  ...args: unknown[]
): Promise<void> {
  // Parallel execution, wait for all to complete
  await Promise.allSettled(
    Array.from(providerHooks.entries()).map(
      ([provider, _]) => invokeHook(provider, hook, ...args)
    )
  );
}
```
