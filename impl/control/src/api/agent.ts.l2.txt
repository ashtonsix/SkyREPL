```typescript
// =============================================================================
// AGENT ENDPOINT ROUTES
// =============================================================================

export function registerAgentRoutes(app: Elysia): void {

  // ─── Heartbeat ───────────────────────────────────────────────────────

  app.post('/v1/agent/heartbeat', async ({ body }) => {
    const hb = body as AgentHeartbeatRequest;

    // Update instance heartbeat timestamp
    await queryOne(
      'UPDATE instances SET last_heartbeat = ? WHERE id = ?',
      [Date.now(), hb.instance_id]
    );

    // Update instance metrics
    if (hb.cpu_percent !== undefined || hb.memory_percent !== undefined) {
      await queryOne(
        `UPDATE instances SET
          cpu_percent = COALESCE(?, cpu_percent),
          memory_percent = COALESCE(?, memory_percent),
          disk_percent = COALESCE(?, disk_percent),
          gpu_utilization = COALESCE(?, gpu_utilization),
          gpu_memory_percent = COALESCE(?, gpu_memory_percent)
         WHERE id = ?`,
        [hb.cpu_percent, hb.memory_percent, hb.disk_percent,
         hb.gpu_utilization, hb.gpu_memory_percent, hb.instance_id]
      );
    }

    // Update Tailscale info
    if (hb.tailscale_ip !== undefined) {
      await queryOne(
        'UPDATE instances SET tailscale_ip = ?, tailscale_status = ? WHERE id = ?',
        [hb.tailscale_ip, hb.tailscale_status ?? null, hb.instance_id]
      );
    }

    // Update SSH session count
    if (hb.active_ssh_sessions !== undefined) {
      await queryOne(
        'UPDATE instances SET active_ssh_sessions = ? WHERE id = ?',
        [hb.active_ssh_sessions, hb.instance_id]
      );
    }

    // Send heartbeat_ack via SSE (if SSE connection exists)
    const hasPendingCommands = await sseManager.hasPendingCommands(hb.instance_id);
    await sseManager.sendCommand(hb.instance_id, {
      type: 'heartbeat_ack',
      control_plane_id: getControlPlaneId(),
    });

    return {
      ack: true,
      control_plane_id: getControlPlaneId(),
      server_time: Date.now(),
      commands_pending: hasPendingCommands,
    };
  });

  // ─── Logs ────────────────────────────────────────────────────────────

  app.post('/v1/agent/logs', async ({ body }) => {
    const log = body as AgentLogsRequest;

    // Store log entry
    await queryOne(
      `INSERT INTO run_logs (run_id, stream, data, timestamp, sequence, phase)
       VALUES (?, ?, ?, ?, ?, ?)`,
      [log.run_id, log.stream, log.data, log.timestamp, log.sequence ?? null, log.phase ?? null]
    );

    // Forward to WebSocket clients watching this run's logs
    broadcastToLogSubscribers(log.run_id, {
      stream: log.stream,
      data: log.data,
      timestamp: log.timestamp,
      sequence: log.sequence,
    });

    // Handle sync_complete: triggers CLAIMED -> ACTIVE transition
    if (log.stream === 'sync_complete') {
      await handleSyncComplete(log.run_id, log.sync_success ?? false);
    }

    return {
      ack: true,
      bytes_received: log.data.length,
    };
  });

  // ─── Artifacts ───────────────────────────────────────────────────────

  app.post('/v1/agent/artifacts', async ({ body, request }) => {
    // Deprecated: prefer presigned URLs
    // Multipart form handling
    const formData = await request.formData();
    const runId = formData.get('run_id') as string;
    const path = formData.get('path') as string;
    const file = formData.get('file') as File;

    const artifactId = generateId();
    const s3Key = `artifacts/${runId}/${path}`;
    const now = Date.now();

    // Upload to S3
    await uploadToS3(s3Key, await file.arrayBuffer());

    // Create blob record for the artifact payload
    const blobResult = await queryOne(
      `INSERT INTO blobs (bucket, checksum, s3_key, s3_bucket, size_bytes, created_at, last_referenced_at)
       VALUES ('artifacts', NULL, ?, ?, ?, ?, ?)`,
      [s3Key, S3_BUCKET, file.size, now, now]
    );
    const blobId = blobResult.lastInsertRowid;

    // Create object record referencing the blob
    await queryOne(
      `INSERT INTO objects (id, type, blob_id, metadata_json, created_at, accessed_at)
       VALUES (?, 'artifact', ?, ?, ?, NULL)`,
      [artifactId, blobId, JSON.stringify({ run_id: runId, path, content_type: file.type }),
       now]
    );

    return {
      artifact_id: artifactId,
      path,
      size_bytes: file.size,
    };
  });

  // ─── Status ──────────────────────────────────────────────────────────

  app.post('/v1/agent/status', async ({ body }) => {
    // Deprecated: prefer log messages (sync_complete, run_complete)
    const status = body as AgentStatusRequest;

    await queryOne(
      `UPDATE runs SET
        workflow_state = ?,
        exit_code = COALESCE(?, exit_code),
        finished_at = CASE WHEN ? IN ('completed', 'failed', 'timeout') THEN ? ELSE finished_at END
       WHERE id = ?`,
      [
        mapStatusToWorkflowState(status.status),
        status.exit_code,
        status.status, Date.now(),
        status.run_id,
      ]
    );

    // Get hold duration for response
    const alloc = await queryOne(
      'SELECT debug_hold_until FROM allocations WHERE run_id = ?',
      [status.run_id]
    );
    const holdDurationMs = alloc?.debug_hold_until
      ? Math.max(0, alloc.debug_hold_until - Date.now())
      : 300000;

    return {
      ack: true,
      hold_duration_ms: holdDurationMs,
    };
  });

  // ─── Spot Interrupt Start ────────────────────────────────────────────

  app.post('/v1/agent/spot-interrupt-start', async ({ body }) => {
    const req = body as SpotInterruptStartRequest;

    logger.warn('Spot interrupt detected', {
      run_id: req.run_id,
      time_remaining_ms: req.time_remaining_ms,
    });

    // Mark run as spot-interrupted
    await queryOne(
      "UPDATE runs SET spot_interrupted = 1, workflow_state = 'spot-interrupt:checkpointing' WHERE id = ?",
      [req.run_id]
    );

    // Notify workflow engine
    broadcastToWorkflowSubscribers(req.run_id, {
      event: 'spot_interrupt',
      data: { run_id: req.run_id, time_remaining_ms: req.time_remaining_ms },
    });

    return { ack: true };
  });

  // ─── Spot Interrupt Complete ─────────────────────────────────────────

  app.post('/v1/agent/spot-interrupt-complete', async ({ body }) => {
    const req = body as SpotInterruptCompleteRequest;

    logger.info('Spot interrupt checkpoint complete', {
      run_id: req.run_id,
      checkpoint_exit_code: req.checkpoint_exit_code,
      artifacts_uploaded: req.artifacts_uploaded,
    });

    // Update run status
    const newState = req.checkpoint_exit_code === 0
      ? 'spot-interrupt:checkpoint-complete'
      : 'spot-interrupt:checkpoint-failed';
    await queryOne(
      'UPDATE runs SET workflow_state = ?, finished_at = ? WHERE id = ?',
      [newState, Date.now(), req.run_id]
    );

    return { ack: true };
  });

  // ─── Heartbeat Panic Start ──────────────────────────────────────────

  app.post('/v1/agent/heartbeat-panic-start', async ({ body }) => {
    const req = body as HeartbeatPanicStartRequest;

    logger.warn('Heartbeat panic detected', {
      run_id: req.run_id,
      time_since_ack_ms: req.time_since_ack_ms,
    });

    await queryOne(
      "UPDATE runs SET workflow_state = 'heartbeat-panic:checkpointing' WHERE id = ?",
      [req.run_id]
    );

    return { ack: true };
  });

  // ─── Heartbeat Panic Complete ────────────────────────────────────────

  app.post('/v1/agent/heartbeat-panic-complete', async ({ body }) => {
    const req = body as HeartbeatPanicCompleteRequest;

    logger.info('Heartbeat panic checkpoint complete', {
      run_id: req.run_id,
      checkpoint_exit_code: req.checkpoint_exit_code,
      artifacts_uploaded: req.artifacts_uploaded,
    });

    // Graceful panic with checkpoint => COMPLETE; no checkpoint => FAILED
    const newState = req.checkpoint_exit_code === 0
      ? 'heartbeat-panic:checkpoint-complete'
      : 'heartbeat-panic:checkpoint-failed';
    await queryOne(
      'UPDATE runs SET workflow_state = ?, finished_at = ? WHERE id = ?',
      [newState, Date.now(), req.run_id]
    );

    return { ack: true };
  });

  // ─── SSE Commands Stream ─────────────────────────────────────────────

  app.get('/v1/agent/commands', async ({ query, request }) => {
    // Handled by sse-protocol.ts
    // See: impl/control/src/api/sse-protocol.ts.l2.txt §SSECommandStream
    return sseManager.createCommandStream(
      query.instance_id as string,
      query.allocation_id as string | undefined,
      request
    );
  });

  // ─── File Download (Deprecated) ─────────────────────────────────────

  app.get('/v1/agent/files/:path', async ({ params, query, set }) => {
    // Deprecated: files now use presigned URLs in start_run message
    const runId = query.run_id as string;
    const checksum = query.checksum as string;

    const file = await queryOne(
      'SELECT * FROM run_files WHERE run_id = ? AND path = ?',
      [runId, params.path]
    );

    if (!file) {
      throw createApiError('RESOURCE_NOT_FOUND', 'File not found', 'not_found');
    }

    if (checksum && file.checksum !== checksum) {
      throw createApiError(
        'INVALID_STATE_TRANSITION',
        'Checksum mismatch (file modified)',
        'conflict'
      );
    }

    // Return file content from S3 or inline storage
    set.headers['content-type'] = getMimeType(params.path);
    set.headers['content-length'] = String(file.size_bytes);
    set.headers['x-checksum'] = file.checksum;

    if (file.s3_key) {
      // Redirect to presigned URL
      const url = await generatePresignedUrl(file.s3_key, 300);
      set.status = 302;
      set.headers['location'] = url;
      return;
    }

    // Inline content
    return file.content;
  });

  // ─── Panic Diagnostics ──────────────────────────────────────────────

  app.post('/v1/instances/:id/panic', async ({ params, body }) => {
    const panic = body as PanicDiagnosticsRequest;

    logger.error('Instance panic received', {
      instance_id: params.id,
      reason: panic.reason,
    });

    // Store panic record for post-mortem analysis
    await queryOne(
      `INSERT INTO instance_panic_logs
       (instance_id, reason, last_state_json, diagnostics_json, error_logs_json, timestamp, created_at)
       VALUES (?, ?, ?, ?, ?, ?, ?)`,
      [
        params.id,
        panic.reason,
        JSON.stringify(panic.last_state),
        JSON.stringify(panic.diagnostics),
        JSON.stringify(panic.error_logs),
        panic.timestamp,
        Date.now(),
      ]
    );

    return { ack: true, stored: true };
  });
}
```

```typescript
/** Handle sync_complete log: trigger CLAIMED -> ACTIVE transition */
async function handleSyncComplete(runId: string, syncSuccess: boolean): Promise<void> {
  if (!syncSuccess) {
    logger.error('File sync failed', { run_id: runId });
    await queryOne(
      "UPDATE runs SET workflow_state = 'launch-run:sync-failed' WHERE id = ?",
      [runId]
    );
    return;
  }

  // Transition allocation CLAIMED -> ACTIVE
  await transaction(async (tx) => {
    const alloc = await queryOne(
      "SELECT * FROM allocations WHERE run_id = ? AND status = 'CLAIMED'",
      [runId]
    );
    if (alloc) {
      await tx.run(
        "UPDATE allocations SET status = 'ACTIVE', updated_at = ? WHERE id = ?",
        [Date.now(), alloc.id]
      );
    }
  });

  logger.info('Sync complete, allocation now ACTIVE', { run_id: runId });
}

function mapStatusToWorkflowState(status: string): string {
  const mapping: Record<string, string> = {
    'started': 'launch-run:running',
    'completed': 'launch-run:complete',

    'timeout': 'launch-run:timeout',

  return mapping[status] ?? `launch-run:${status}`;

function getMimeType(path: string): string {
  const ext = path.split('.').pop()?.toLowerCase();
  const types: Record<string, string> = {
    'py': 'text/x-python',
    'js': 'application/javascript',
    'ts': 'application/typescript',
    'json': 'application/json',
    'txt': 'text/plain',
```

```sql
-- =============================================================================
-- PANIC STORAGE SCHEMA
-- =============================================================================

CREATE TABLE instance_panic_logs (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    instance_id INTEGER NOT NULL,
    reason TEXT NOT NULL,
    last_state_json TEXT NOT NULL,
    diagnostics_json TEXT NOT NULL,
    error_logs_json TEXT NOT NULL,
    timestamp INTEGER NOT NULL,
    created_at INTEGER NOT NULL,

    FOREIGN KEY (instance_id) REFERENCES instances(id)
);

CREATE INDEX idx_panic_instance ON instance_panic_logs(instance_id);
CREATE INDEX idx_panic_timestamp ON instance_panic_logs(timestamp);
```
